    ```typescript
    type Params = { params: Promise<{ slug: string }> };

    export async function generateMetadata({ params }: Params): Promise<Metadata> {
      const { slug } = await params;
      const agency = await fetchAgentBySlug(slug);
      if (!agency) return { title: "Agency Not Found | Britestate" };
      return {
        title: `${agency.agency_name} Estate Agents | Britestate`,
        description: agency.description ?? `View ${agency.agency_name}'s listings, sold properties, team and reviews on Britestate.`,
        openGraph: {
          title: agency.agency_name,
          type: "profile",
          images: agency.logo_url ? [{ url: agency.logo_url }] : [],
        },
      };
    }

    export default async function AgentProfilePage({ params }: Params) {
      const { slug } = await params;
      const agency = await fetchAgentBySlug(slug);
      if (!agency) notFound();

      const [stats, activeListings, soldLetListings] = await Promise.all([
        fetchAgentStats(slug),
        fetchAgentListings(agency.user_id, "active", 0),
        fetchAgentListings(agency.user_id, "sold_let", 0),
      ]);

      const jsonLd = buildAgentJsonLd(agency, stats); // additive export from jsonld.ts

      return (
        <>
          <script type="application/ld+json" dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }} />
          <div className="min-h-screen bg-slate-50 dark:bg-slate-950">
            <main className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-8">
              <AgencyHero agency={agency} stats={stats} />
              <div className="grid grid-cols-1 lg:grid-cols-12 gap-8">
                <div className="lg:col-span-8">
                  <AgentProfileTabs
                    activeListingsCount={stats.active_listings_count}
                    soldLetCount={stats.sold_count}
                    listings={<ListingsTab listings={activeListings.listings} total={activeListings.total} />}
                    soldLet={<SoldLetTab listings={soldLetListings.listings} total={soldLetListings.total} />}
                    reviews={<div className="text-slate-500 text-sm py-4">Reviews — Plan 17-05</div>}
                    team={<div className="text-slate-500 text-sm py-4">Team — Plan 17-05</div>}
                  />
                </div>
                <div className="lg:col-span-4">
                  <AgentSidebar agency={agency} />
                </div>
              </div>
            </main>
          </div>
        </>
      );
    }
    ```
