**src/lib/yield-calculator.ts** — pure functions (no Supabase, no React):
```typescript
export type YieldInputs = {
  propertyValue: number;   // purchase price or current value
  monthlyRent: number;
  monthlyManagementFee: number;  // % of rent typically, entered as £/month
  monthlyMaintenance: number;
  monthlyInsurance: number;
  monthlyMortgage: number;       // interest portion only for accurate net yield
};
export type YieldResult = {
  grossYield: number;  // as percentage, rounded to 2dp
  netYield: number;    // as percentage, rounded to 2dp
  annualRent: number;
  annualCosts: number;
  annualNet: number;
};

export function calculateYield(inputs: YieldInputs): YieldResult {
  if (inputs.propertyValue <= 0) return { grossYield: 0, netYield: 0, annualRent: 0, annualCosts: 0, annualNet: 0 };
  const annualRent = inputs.monthlyRent * 12;
  const annualCosts = (inputs.monthlyManagementFee + inputs.monthlyMaintenance + inputs.monthlyInsurance + inputs.monthlyMortgage) * 12;
  const grossYield = parseFloat(((annualRent / inputs.propertyValue) * 100).toFixed(2));
  const netYield = parseFloat((((annualRent - annualCosts) / inputs.propertyValue) * 100).toFixed(2));
  return { grossYield, netYield, annualRent, annualCosts, annualNet: annualRent - annualCosts };
}
```
