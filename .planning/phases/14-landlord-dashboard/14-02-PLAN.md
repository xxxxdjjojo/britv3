```typescript
export type Section21ValidationResult = { valid: boolean; errors: string[] };

export function validateSection21Requirements(notice: Partial<LegalNotice>): Section21ValidationResult {
  const errors: string[] = [];
  if (!notice.epc_provided) errors.push("EPC must have been provided to tenant");
  if (!notice.gas_safety_provided) errors.push("Gas Safety Certificate must have been provided within 28 days");
  if (!notice.deposit_scheme_reference?.trim()) errors.push("Deposit protection scheme reference is required");
  if (!notice.possession_date) errors.push("Possession date is required");
  return { valid: errors.length === 0, errors };
}
```
