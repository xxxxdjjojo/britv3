-- Valid state transitions
CREATE TABLE booking_state_transitions (
  id SERIAL PRIMARY KEY,
  from_status booking_status NOT NULL,
  to_status booking_status NOT NULL,
  allowed_by TEXT[] NOT NULL, -- ['user', 'provider', 'system']
  requires_reason BOOLEAN DEFAULT FALSE,
  
  UNIQUE(from_status, to_status)
);

-- Define valid transitions
INSERT INTO booking_state_transitions (from_status, to_status, allowed_by, requires_reason) VALUES
  ('pending_confirmation', 'confirmed', ARRAY['provider'], FALSE),
  ('pending_confirmation', 'cancelled', ARRAY['user', 'provider'], TRUE),
  ('confirmed', 'in_progress', ARRAY['provider'], FALSE),
  ('confirmed', 'cancelled', ARRAY['user', 'provider'], TRUE),
  ('in_progress', 'completed', ARRAY['provider'], FALSE),
  ('in_progress', 'cancelled', ARRAY['provider'], TRUE),
  ('completed', 'disputed', ARRAY['user'], TRUE),
  ('cancelled', 'confirmed', ARRAY['provider'], FALSE), -- Rebooking scenario
  ('disputed', 'completed', ARRAY['system'], TRUE); -- After resolution

-- State transition validation function
CREATE OR REPLACE FUNCTION validate_booking_state_transition(
  p_booking_id UUID,
  p_new_status booking_status,
  p_user_id UUID,
  p_reason TEXT DEFAULT NULL
)
RETURNS JSONB AS $$
DECLARE
  v_booking RECORD;
  v_transition RECORD;
  v_user_role TEXT;
BEGIN
  -- Get current booking state
  SELECT b.*, l.provider_id, sr.user_id as requester_id
  INTO v_booking
  FROM bookings b
  JOIN quotes q ON b.quote_id = q.id
  JOIN service_requests sr ON q.service_request_id = sr.id
  JOIN listings l ON l.id = q.service_request_id -- Assuming listings table exists
  WHERE b.id = p_booking_id;
  
  IF NOT FOUND THEN
    RETURN jsonb_build_object('valid', false, 'error', 'Booking not found');
  END IF;
  
  -- Determine user role in this booking
  IF p_user_id = v_booking.provider_id THEN
    v_user_role := 'provider';
  ELSIF p_user_id = v_booking.requester_id THEN
    v_user_role := 'user';
  ELSE
    v_user_role := 'system'; -- For admin/automated transitions
  END IF;
  
  -- Check if transition is valid
  SELECT * INTO v_transition
  FROM booking_state_transitions
  WHERE from_status = v_booking.status
    AND to_status = p_new_status
    AND v_user_role = ANY(allowed_by);
  
  IF NOT FOUND THEN
    RETURN jsonb_build_object(
      'valid', false, 
      'error', format('Invalid transition from %s to %s for role %s', 
                     v_booking.status, p_new_status, v_user_role)
    );
  END IF;
  
  -- Check reason requirement
  IF v_transition.requires_reason AND (p_reason IS NULL OR LENGTH(p_reason) < 10) THEN
    RETURN jsonb_build_object(
      'valid', false,
      'error', 'Reason required for this state transition (min 10 characters)'
    );
  END IF;
  
  RETURN jsonb_build_object(
    'valid', true,
    'from_status', v_booking.status,
    'to_status', p_new_status,
    'user_role', v_user_role
  );
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER;

-- Safe state transition function
CREATE OR REPLACE FUNCTION transition_booking_status(
  p_booking_id UUID,
  p_new_status booking_status,
  p_user_id UUID,
  p_reason TEXT DEFAULT NULL
)
RETURNS JSONB AS $$
DECLARE
  v_validation JSONB;
  v_old_status booking_status;
BEGIN
  -- Validate transition
  v_validation := validate_booking_state_transition(
    p_booking_id, p_new_status, p_user_id, p_reason
  );
  
  IF NOT (v_validation->>'valid')::BOOLEAN THEN
    RETURN v_validation;
  END IF;
  
  -- Get current status for history
  SELECT status INTO v_old_status
  FROM bookings
  WHERE id = p_booking_id;
  
  -- Perform transition
  UPDATE bookings
  SET 
    status = p_new_status,
    confirmed_at = CASE WHEN p_new_status = 'confirmed' THEN NOW() ELSE confirmed_at END,
    completed_at = CASE WHEN p_new_status = 'completed' THEN NOW() ELSE completed_at END,
    cancelled_at = CASE WHEN p_new_status = 'cancelled' THEN NOW() ELSE cancelled_at END,
    cancellation_reason = CASE WHEN p_new_status = 'cancelled' THEN p_reason ELSE cancellation_reason END,
    cancelled_by = CASE WHEN p_new_status = 'cancelled' THEN p_user_id ELSE cancelled_by END,
    updated_at = NOW()
  WHERE id = p_booking_id;
  
  -- Log status change
  INSERT INTO booking_status_history (
    booking_id,
    from_status,
    to_status,
    changed_by,
    reason,
    created_at
  ) VALUES (
    p_booking_id,
    v_old_status,
    p_new_status,
    p_user_id,
    p_reason,
    NOW()
  );
  
  RETURN jsonb_build_object(
    'valid', true,
    'booking_id', p_booking_id,
    'new_status', p_new_status,
    'message', format('Booking transitioned from %s to %s', v_old_status, p_new_status)
  );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Booking status history table
CREATE TABLE booking_status_history (
  id BIGSERIAL PRIMARY KEY,
  booking_id UUID NOT NULL REFERENCES bookings(id) ON DELETE CASCADE,

CREATE INDEX idx_booking_status_history_user ON booking_status_history(changed_by, created_at DESC);

-- Function to check for booking conflicts
CREATE OR REPLACE FUNCTION check_booking_conflicts(
  p_provider_id UUID,
  p_start_date DATE,
  p_end_date DATE,
  p_exclude_booking_id UUID DEFAULT NULL
)
RETURNS TABLE (
  has_conflict BOOLEAN,
  conflicting_bookings UUID[]
) AS $$
DECLARE
  v_conflicts UUID[];
BEGIN
  -- Find overlapping bookings for this provider
  SELECT ARRAY_AGG(id)
  INTO v_conflicts
  FROM bookings
  WHERE provider_id = p_provider_id
    AND status IN ('confirmed', 'in_progress')
    AND (p_exclude_booking_id IS NULL OR id != p_exclude_booking_id)
    AND (
      -- Check for date range overlap
      (scheduled_start_date <= p_end_date AND scheduled_end_date >= p_start_date)
      OR
      (p_start_date <= scheduled_end_date AND p_end_date >= scheduled_start_date)
    );
  
  RETURN QUERY SELECT 
    (v_conflicts IS NOT NULL AND array_length(v_conflicts, 1) > 0),
    COALESCE(v_conflicts, ARRAY[]::UUID[]);
END;
$$ LANGUAGE plpgsql STABLE;

-- Trigger to prevent overlapping bookings
CREATE OR REPLACE FUNCTION prevent_booking_conflicts()
RETURNS TRIGGER AS $$
DECLARE
  v_conflict_check RECORD;
BEGIN
  -- Only check for confirmed/in-progress bookings
  IF NEW.status NOT IN ('confirmed', 'in_progress') THEN
    RETURN NEW;
  END IF;
  
  -- Skip if no dates set
  IF NEW.scheduled_start_date IS NULL OR NEW.scheduled_end_date IS NULL THEN
    RETURN NEW;
  END IF;
  
  -- Check for conflicts
  SELECT * INTO v_conflict_check
  FROM check_booking_conflicts(
    NEW.provider_id,
    NEW.scheduled_start_date,
    NEW.scheduled_end_date,
    NEW.id
  );
  
  IF v_conflict_check.has_conflict THEN
    RAISE EXCEPTION 'Booking conflict detected. Provider already has confirmed bookings for these dates: %',
      v_conflict_check.conflicting_bookings
      USING ERRCODE = 'P0001';
  END IF;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER enforce_booking_no_conflicts
  BEFORE INSERT OR UPDATE ON bookings
  FOR EACH ROW
  EXECUTE FUNCTION prevent_booking_conflicts();

-- Provider availability calendar
CREATE TABLE provider_availability (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  provider_id UUID NOT NULL REFERENCES service_provider_details(user_id) ON DELETE CASCADE,
  
  -- Date range
  start_date DATE NOT NULL,
  end_date DATE NOT NULL,
  
  -- Availability status
  is_available BOOLEAN DEFAULT TRUE,
  reason TEXT, -- e.g., "Holiday", "Maintenance", "Booked"
  
  -- Recurring patterns (optional)
  recurrence_rule TEXT, -- iCalendar RRULE format
  
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  
  CONSTRAINT valid_date_range CHECK (start_date <= end_date)
);

CREATE INDEX idx_provider_availability_dates ON provider_availability(provider_id, start_date, end_date);
CREATE INDEX idx_provider_availability_status ON provider_availability(provider_id, is_available);

-- Function to check provider availability
CREATE OR REPLACE FUNCTION is_provider_available(
  p_provider_id UUID,
  p_start_date DATE,
  p_end_date DATE
)
RETURNS BOOLEAN AS $$
DECLARE
  v_unavailable_count INTEGER;
BEGIN
  -- Check for unavailable periods
  SELECT COUNT(*)
  INTO v_unavailable_count
  FROM provider_availability
  WHERE provider_id = p_provider_id
    AND is_available = FALSE
    AND (
      (start_date <= p_end_date AND end_date >= p_start_date)
    );
  
  RETURN v_unavailable_count = 0;
END;
$$ LANGUAGE plpgsql STABLE;

typescriptimport { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { emailQueue, defaultJobOptions } from '@/lib/queue/config';
import { z } from 'zod';

const createBookingSchema = z.object({
  quote_id: z.string().uuid(),
  scheduled_start_date: z.string().optional(), // ISO date
  scheduled_end_date: z.string().optional(), // ISO date
  user_notes: z.string().max(1000).optional(),
});

export async function POST(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    // Parse request
    const body = await request.json();
    const data = createBookingSchema.parse(body);
    
    // Get quote details
    const { data: quote, error: quoteError } = await supabase
      .from('quotes')
      .select(`
        *,
        service_request:service_requests!inner(
          id,
          user_id,
          title,
          property_address
        )
      `)
      .eq('id', data.quote_id)
      .single();
    
    if (quoteError || !quote) {
      return NextResponse.json({ error: 'Quote not found' }, { status: 404 });
    }
    
    // Verify user owns the service request
    if (quote.service_request.user_id !== user.id) {
      return NextResponse.json(
        { error: 'You can only create bookings for your own service requests' },
        { status: 403 }
      );
    }
    
    // Verify quote is accepted
    if (quote.status !== 'accepted') {
      return NextResponse.json(
        { error: 'Quote must be accepted before creating a booking' },
        { status: 400 }
      );
    }
    
    // Check for existing booking
    const { data: existingBooking } = await supabase
      .from('bookings')
      .select('id, status')
      .eq('quote_id', data.quote_id)
      .single();
    
    if (existingBooking) {
      if (existingBooking.status !== 'cancelled') {
        return NextResponse.json(
          { error: 'Booking already exists for this quote' },
          { status: 400 }
        );
      }
    }
    
    // Check provider availability if dates provided
    if (data.scheduled_start_date && data.scheduled_end_date) {
      const { data: availabilityCheck } = await supabase.rpc('is_provider_available', {
        p_provider_id: quote.provider_id,
        p_start_date: data.scheduled_start_date,
        p_end_date: data.scheduled_end_date,
      });
      
      if (!availabilityCheck) {
        return NextResponse.json(
          { error: 'Provider is not available for the selected dates' },
          { status: 409 }
        );
      }
    }
    
    // Create booking
    const { data: booking, error: bookingError } = await supabase
      .from('bookings')
      .insert({
        service_request_id: quote.service_request_id,
        quote_id: data.quote_id,
        user_id: user.id,
        provider_id: quote.provider_id,
        scheduled_start_date: data.scheduled_start_date || null,
        scheduled_end_date: data.scheduled_end_date || null,
        estimated_duration_hours: quote.estimated_duration_hours,
        user_notes: data.user_notes || null,
        status: 'pending_confirmation',
      })
      .select()
      .single();
    
    if (bookingError) {
      console.error('Booking creation error:', bookingError);
      
      // Check if it's a conflict error
      if (bookingError.code === 'P0001') {
        return NextResponse.json(
          { error: 'Booking conflict: Provider already has confirmed bookings for these dates' },
          { status: 409 }
        );
      }
      
      return NextResponse.json(
        { error: 'Failed to create booking' },
        { status: 500 }
      );
    }
    
    // Get provider details for notification
    const { data: providerDetails } = await supabase
      .from('service_provider_details')
      .select('business_name, user_id')
      .eq('user_id', quote.provider_id)
      .single();
    
    const { data: providerUser } = await supabase
      .from('auth.users')
      .select('email')
      .eq('id', quote.provider_id)
      .single();
    
    // Queue notification to provider
    await emailQueue.add(
      'booking-created',
      {
        type: 'booking_created',
        data: {
          to: providerUser?.email,
          providerName: providerDetails?.business_name || 'Provider',
          bookingReference: booking.booking_reference,
          serviceTitle: quote.service_request.title,
          customerName: user.email, // TODO: Get from profile
          scheduledDate: data.scheduled_start_date || 'To be confirmed',
          viewUrl: `${process.env.NEXT_PUBLIC_APP_URL}/provider/bookings/${booking.id}`,
        },
      },
      defaultJobOptions
    );
    
    return NextResponse.json({
      success: true,
      booking: {
        id: booking.id,
        booking_reference: booking.booking_reference,
        status: booking.status,
        scheduled_start_date: booking.scheduled_start_date,
        scheduled_end_date: booking.scheduled_end_date,
      },
      processing_time_ms: Date.now() - startTime,
    });
    
  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: 'Invalid request data', details: error.errors },
        { status: 400 }
      );
    }
    
    console.error('Booking creation error:', error);
    return NextResponse.json(
      { error: 'Internal server error' },
      { status: 500 }
    );
  }
}
Update Booking Status Endpoint
File: app/api/bookings/[id]/status/route.ts
typescriptimport { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { emailQueue, defaultJobOptions } from '@/lib/queue/config';
import { z } from 'zod';

const updateStatusSchema = z.object({
  status: z.enum([
    'pending_confirmation',
    'confirmed',
    'in_progress',
    'completed',
    'cancelled',
    'disputed'
  ]),
  reason: z.string().min(10).max(500).optional(),
  actual_start_date: z.string().optional(), // ISO date
  actual_end_date: z.string().optional(), // ISO date
});

export async function PATCH(
  request: NextRequest,
  { params }: { params: { id: string } }
) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    const bookingId = params.id;
    const body = await request.json();
    const data = updateStatusSchema.parse(body);
    
    // Get booking details
    const { data: booking, error: bookingError } = await supabase
      .from('bookings')
      .select(`
        *,
        service_request:service_requests!inner(title),
        quote:quotes!inner(total_amount)
      `)
      .eq('id', bookingId)
      .single();
    
    if (bookingError || !booking) {
      return NextResponse.json({ error: 'Booking not found' }, { status: 404 });
    }
    
    // Verify user is involved in this booking
    if (booking.user_id !== user.id && booking.provider_id !== user.id) {
      return NextResponse.json(
        { error: 'You do not have permission to update this booking' },
        { status: 403 }
      );
    }
    
    // Validate state transition
    const { data: validationResult } = await supabase.rpc(
      'validate_booking_state_transition',
      {
        p_booking_id: bookingId,
        p_new_status: data.status,
        p_user_id: user.id,
        p_reason: data.reason || null,
      }
    );
    
    if (!validationResult?.valid) {
      return NextResponse.json(
        { error: validationResult?.error || 'Invalid status transition' },
        { status: 400 }
      );
    }
    
    // Perform transition
    const { data: transitionResult } = await supabase.rpc(
      'transition_booking_status',
      {
        p_booking_id: bookingId,
        p_new_status: data.status,
        p_user_id: user.id,
        p_reason: data.reason || null,
      }
    );
    
    if (!transitionResult?.valid) {
      return NextResponse.json(
        { error: transitionResult?.error || 'Status transition failed' },
        { status: 400 }
      );
    }
    
    // Update actual dates if provided
    if (data.actual_start_date || data.actual_end_date) {
      await supabase
        .from('bookings')
        .update({
          actual_start_date: data.actual_start_date || undefined,
          actual_end_date: data.actual_end_date || undefined,
        })
        .eq('id', bookingId);
    }
    
    // Send notification based on new status
    await sendStatusChangeNotification(
      supabase,
      booking,
      data.status,
      user.id
    );
    
    return NextResponse.json({
      success: true,
      booking_id: bookingId,
      new_status: data.status,
      message: transitionResult.message,
      processing_time_ms: Date.now() - startTime,
    });
    
  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: 'Invalid request data', details: error.errors },
        { status: 400 }
      );
    }
    
    console.error('Status update error:', error);
    return NextResponse.json(
      { error: 'Internal server error' },
      { status: 500 }
    );
  }
}

async function sendStatusChangeNotification(
  supabase: any,
  booking: any,
  newStatus: string,
  triggeredBy: string
) {
  const recipientId = triggeredBy === booking.user_id 
    ? booking.provider_id 
    : booking.user_id;
  
  const { data: recipient } = await supabase
    .from('auth.users')
    .select('email')
    .eq('id', recipientId)
    .single();
  
  if (!recipient) return;
  
  let emailType = '';
  let emailData: any = {
    to: recipient.email,
    bookingReference: booking.booking_reference,
    serviceTitle: booking.service_request.title,
  };
  
  switch (newStatus) {
    case 'confirmed':
      emailType = 'booking_confirmed';
      emailData.scheduledDate = booking.scheduled_start_date;
      break;
    case 'in_progress':
      emailType = 'booking_started';
      break;
    case 'completed':
      emailType = 'booking_completed';
      emailData.reviewUrl = `${process.env.NEXT_PUBLIC_APP_URL}/bookings/${booking.id}/review`;
      break;
    case 'cancelled':
      emailType = 'booking_cancelled';
      emailData.reason = booking.cancellation_reason;
      break;
    default:
      return;
  }
  
  await emailQueue.add(
    emailType,
    {
      type: emailType,
      data: emailData,
    },
    defaultJobOptions
  );
}

typescriptimport { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { Redis } from '@upstash/redis';

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!,
});

export async function GET(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    const { searchParams } = new URL(request.url);
    const status = searchParams.get('status');
    const role = searchParams.get('role'); // 'customer' or 'provider'
    const page = parseInt(searchParams.get('page') || '1');
    const perPage = 20;
    
    // Check cache
    const cacheKey = `bookings:list:${user.id}:${status}:${role}:${page}`;
    const cached = await redis.get(cacheKey);
    
    if (cached) {
      return NextResponse.json({
        ...cached,
        cached: true,
        query_time_ms: Date.now() - startTime,
      });
    }
    
    // Determine which bookings to fetch
    let query = supabase
      .from('bookings')
      .select(`
        id,
        booking_reference,
        status,
        scheduled_start_date,
        scheduled_end_date,
        actual_start_date,
        actual_end_date,
        created_at,
        service_request:service_requests!inner(
          id,
          title,
          service_category,
          property_postcode
        ),
        quote:quotes!inner(
          total_amount,
          currency
        ),
        provider:service_provider_details!inner(
          business_name,
          user_id
        )
      `, { count: 'exact' })
      .order('created_at', { ascending: false });
    
    // Filter by role
    if (role === 'provider') {
      query = query.eq('provider_id', user.id);
    } else {
      query = query.eq('user_id', user.id);
    }
    
    // Filter by status
    if (status) {
      query = query.eq('status', status);
    }
    
    // Pagination
    const offset = (page - 1) * perPage;
    query = query.range(offset, offset + perPage - 1);
    
    const { data: bookings, error, count } = await query;
    
    if (error) {
      throw error;
    }
    
    // Calculate stats
    const { data: stats } = await supabase
      .from('bookings')
      .select('status', { count: 'exact', head: true })
      .eq(role === 'provider' ? 'provider_id' : 'user_id', user.id);
    
    const response = {
      bookings: bookings || [],
      pagination: {
        page,
        per_page: perPage,
        total: count || 0,
        total_pages: Math.ceil((count || 0) / perPage),
      },
      stats: {
        // TODO: Calculate from database
        pending: 0,
        confirmed: 0,
        in_progress: 0,
        completed: 0,
      },
      query_time_ms: Date.now() - startTime,
    };
    
    // Cache for 30 seconds
    await redis.setex(cacheKey, 30, response);
    
    return NextResponse.json(response);
    
  } catch (error) {
    console.error('Booking list error:', error);
    return NextResponse.json(
      { error: 'Failed to fetch bookings' },
      { status: 500 }
    );
  }
}
