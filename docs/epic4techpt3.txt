CREATE MATERIALIZED VIEW provider_rating_stats AS
SELECT 
  r.provider_id,
  
  -- Overall ratings
  ROUND(AVG(r.overall_rating)::NUMERIC, 2) AS average_rating,
  STDDEV(r.overall_rating) AS rating_stddev,
  COUNT(r.id) AS total_reviews,
  
  -- Individual dimensions
  ROUND(AVG(r.punctuality_rating)::NUMERIC, 2) AS avg_punctuality,
  ROUND(AVG(r.quality_rating)::NUMERIC, 2) AS avg_quality,
  ROUND(AVG(r.value_rating)::NUMERIC, 2) AS avg_value,
  ROUND(AVG(r.professionalism_rating)::NUMERIC, 2) AS avg_professionalism,
  
  -- Distribution
  SUM(CASE WHEN r.overall_rating = 5 THEN 1 ELSE 0 END) AS count_5_star,
  SUM(CASE WHEN r.overall_rating = 4 THEN 1 ELSE 0 END) AS count_4_star,
  SUM(CASE WHEN r.overall_rating = 3 THEN 1 ELSE 0 END) AS count_3_star,
  SUM(CASE WHEN r.overall_rating = 2 THEN 1 ELSE 0 END) AS count_2_star,
  SUM(CASE WHEN r.overall_rating = 1 THEN 1 ELSE 0 END) AS count_1_star,
  
  -- Engagement metrics
  SUM(r.helpful_count) AS total_helpful_votes,
  COUNT(CASE WHEN r.provider_response IS NOT NULL THEN 1 END) AS reviews_with_responses,
  ROUND(100.0 * COUNT(CASE WHEN r.provider_response IS NOT NULL THEN 1 END) / NULLIF(COUNT(*), 0), 2) AS response_rate,
  
  -- Recency
  MAX(r.created_at) AS last_review_date,
  AVG((EXTRACT(EPOCH FROM (NOW() - r.created_at)) / 86400))::INTEGER AS avg_days_since_review,
  
  -- Moderation metrics
  SUM(CASE WHEN r.moderation_status = 'approved' THEN 1 ELSE 0 END) AS approved_reviews,
  SUM(CASE WHEN r.moderation_status = 'flagged_for_review' THEN 1 ELSE 0 END) AS flagged_reviews,
  
  -- Sentiment analysis
  SUM(CASE WHEN r.detected_sentiment = 'very_positive' THEN 1 ELSE 0 END) AS sentiment_very_positive,
  SUM(CASE WHEN r.detected_sentiment = 'positive' THEN 1 ELSE 0 END) AS sentiment_positive,
  SUM(CASE WHEN r.detected_sentiment = 'neutral' THEN 1 ELSE 0 END) AS sentiment_neutral,
  SUM(CASE WHEN r.detected_sentiment = 'negative' THEN 1 ELSE 0 END) AS sentiment_negative,
  SUM(CASE WHEN r.detected_sentiment = 'very_negative' THEN 1 ELSE 0 END) AS sentiment_very_negative,
  
  NOW() AS last_updated

FROM reviews r
WHERE r.deleted_at IS NULL
  AND r.moderation_status = 'approved'
GROUP BY r.provider_id;

-- Calculate review authenticity score (fraud detection)
CREATE OR REPLACE FUNCTION calculate_authenticity_score(
  p_review_id UUID,
  p_reviewer_id UUID,
  p_provider_id UUID,
  p_rating INTEGER,
  p_review_text TEXT,
  p_booking_exists BOOLEAN
)
RETURNS NUMERIC AS $$
DECLARE
  v_score NUMERIC := 1.0;
  v_user_review_count INTEGER;
  v_time_since_booking_days INTEGER;
  v_review_length INTEGER;
  v_all_caps_ratio NUMERIC;
  v_extreme_rating_ratio NUMERIC;
  v_similar_reviews INTEGER;
BEGIN
  
  -- Factor 1: User review history (10% penalty per review if many unverified)
  SELECT COUNT(*) INTO v_user_review_count
  FROM reviews
  WHERE user_id = p_reviewer_id AND is_verified_booking = FALSE;
  
  v_score := v_score * (1.0 - (LEAST(v_user_review_count::NUMERIC / 50, 0.5)));
  
  -- Factor 2: Review timing vs booking
  SELECT EXTRACT(DAY FROM (NOW() - completed_at))::INTEGER INTO v_time_since_booking_days
  FROM bookings
  WHERE id IN (
    SELECT booking_id FROM reviews WHERE id = p_review_id
  )
  LIMIT 1;
  
  -- Suspicious if reviewed very quickly or very late
  IF v_time_since_booking_days IS NOT NULL THEN
    IF v_time_since_booking_days < 1 OR v_time_since_booking_days > 365 THEN
      v_score := v_score * 0.8;
    END IF;
  END IF;
  
  -- Factor 3: Review length
  v_review_length := LENGTH(p_review_text);
  IF v_review_length < 10 THEN
    v_score := v_score * 0.7;
  ELSIF v_review_length > 5000 THEN
    v_score := v_score * 0.85;
  END IF;
  
  -- Factor 4: ALL CAPS ratio
  v_all_caps_ratio := REGEXP_MATCHES(p_review_text, '[A-Z]', 'g')::TEXT[] FILTER (WHERE TRUE) IS NOT NULL;
  IF v_all_caps_ratio > 0.5 THEN
    v_score := v_score * 0.8;
  END IF;
  
  -- Factor 5: Extreme ratings (all 5s or all 1s)
  SELECT COUNT(*) INTO v_extreme_rating_ratio
  FROM reviews
  WHERE user_id = p_reviewer_id
    AND (overall_rating = 5 OR overall_rating = 1);
  
  IF v_extreme_rating_ratio > 5 THEN
    v_score := v_score * 0.7;
  END IF;
  
  -- Factor 6: Similar reviews by same user
  SELECT COUNT(*) INTO v_similar_reviews
  FROM reviews r1
  WHERE r1.user_id = p_reviewer_id
    AND r1.id != p_review_id
    AND r1.overall_rating = p_rating
    AND SIMILARITY(r1.review_text, p_review_text) > 0.8;
  
  IF v_similar_reviews > 0 THEN
    v_score := v_score * (1.0 - (v_similar_reviews::NUMERIC * 0.1));
  END IF;
  
  -- Verified booking gets boost
  IF p_booking_exists THEN
    v_score := v_score * 1.2;
  END IF;
  
  RETURN ROUND(LEAST(v_score, 1.0), 2);
END;
$$ LANGUAGE plpgsql;

-- Calculate moderation priority
CREATE OR REPLACE FUNCTION calculate_moderation_priority(
  p_review_id UUID
)
RETURNS INTEGER AS $$
DECLARE
  v_priority INTEGER := 0;
  v_review RECORD;
  v_flag_count INTEGER;
BEGIN
  
  SELECT r.*, f.is_valid
  INTO v_review
  FROM reviews r
  LEFT JOIN review_flags f ON r.id = f.review_id
  WHERE r.id = p_review_id
  LIMIT 1;
  
  -- Base priority
  IF v_review.fake_review_probability > 0.8 THEN
    v_priority := v_priority + 10;
  ELSIF v_review.fake_review_probability > 0.5 THEN
    v_priority := v_priority + 5;
  END IF;
  
  -- Flagged reviews
  SELECT COUNT(*) INTO v_flag_count
  FROM review_flags
  WHERE review_id = p_review_id;
  
  v_priority := v_priority + (v_flag_count * 2);
  
  -- Extreme ratings
  IF v_review.overall_rating IN (1, 5) THEN
    v_priority := v_priority + 1;
  END IF;
  
  RETURN v_priority;
END;
$$ LANGUAGE plpgsql;

-- Search reviews with ranking
CREATE OR REPLACE FUNCTION search_reviews(
  p_provider_id UUID,
  p_search_query TEXT DEFAULT NULL,
  p_min_rating INTEGER DEFAULT NULL,
  p_max_rating INTEGER DEFAULT NULL,
  p_sort_by TEXT DEFAULT 'recent',
  p_limit INTEGER DEFAULT 20,
  p_offset INTEGER DEFAULT 0
)
RETURNS TABLE (
  review_id UUID,
  overall_rating INTEGER,
  review_title TEXT,
  review_text TEXT,
  helpful_count INTEGER,
  created_at TIMESTAMPTZ,
  search_rank REAL
) AS $$
BEGIN
  RETURN QUERY
  SELECT 
    r.id,
    r.overall_rating,
    r.review_title,
    r.review_text,
    r.helpful_count,
    r.created_at,
    CASE
      WHEN p_search_query IS NOT NULL THEN
        ts_rank(r.search_vector, to_tsquery('review_search', p_search_query))
      ELSE 0
    END as search_rank
  FROM reviews r
  WHERE 
    r.provider_id = p_provider_id
    AND r.deleted_at IS NULL
    AND r.moderation_status = 'approved'
    AND (p_min_rating IS NULL OR r.overall_rating >= p_min_rating)
    AND (p_max_rating IS NULL OR r.overall_rating <= p_max_rating)
    AND (
      p_search_query IS NULL 
      OR r.search_vector @@ to_tsquery('review_search', p_search_query)
    )
  ORDER BY
    CASE
      WHEN p_sort_by = 'helpful' THEN r.helpful_count
      WHEN p_sort_by = 'recent' THEN EXTRACT(EPOCH FROM r.created_at)
      WHEN p_sort_by = 'relevant' AND p_search_query IS NOT NULL THEN
        ts_rank(r.search_vector, to_tsquery('review_search', p_search_query))
      ELSE EXTRACT(EPOCH FROM r.created_at)
    END DESC,
    r.created_at DESC
  LIMIT p_limit
  OFFSET p_offset;
END;
$$ LANGUAGE plpgsql STABLE;

-- Automatically update materialized view on review approval
CREATE OR REPLACE FUNCTION on_review_approved()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.moderation_status = 'approved' AND OLD.moderation_status != 'approved' THEN
    PERFORM refresh_provider_rating_stats();
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER review_approval_trigger
  AFTER UPDATE ON reviews
  FOR EACH ROW
  WHEN (NEW.moderation_status IS DISTINCT FROM OLD.moderation_status)
  EXECUTE FUNCTION on_review_approved();

-- Auto-calculate authenticity on review insert
CREATE OR REPLACE FUNCTION on_review_created()
RETURNS TRIGGER AS $$
DECLARE
  v_booking_exists BOOLEAN;
BEGIN
  -- Check if booking is verified
  SELECT EXISTS(
    SELECT 1 FROM bookings 
    WHERE id = NEW.booking_id AND status = 'completed'
  ) INTO v_booking_exists;
  
  NEW.is_verified_booking := v_booking_exists;
  
  -- Calculate authenticity
  NEW.review_authenticity_score := calculate_authenticity_score(
    NEW.id,
    NEW.user_id,
    NEW.provider_id,
    NEW.overall_rating,
    NEW.review_text,
    v_booking_exists
  );
  
  -- Set fake review probability inversely
  NEW.fake_review_probability := ROUND(1.0 - NEW.review_authenticity_score, 2);
  
  -- Calculate initial priority
  NEW.moderation_queue_priority := calculate_moderation_priority(NEW.id);
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER review_created_trigger
  BEFORE INSERT ON reviews
  FOR EACH ROW
  EXECUTE FUNCTION on_review_created();

export async function analyzeReviewSentiment(
  reviewText: string,
  reviewTitle: string
): Promise<SentimentAnalysis> {
  const prompt = `Analyze this review for sentiment, key topics, and concerns.

Review Title: ${reviewTitle}
Review Text: ${reviewText}

Respond ONLY with valid JSON in this exact format:
{
  "sentiment": "very_positive" | "positive" | "neutral" | "negative" | "very_negative",
  "confidence": 0.85,
  "keywords": ["keyword1", "keyword2"],
  "topics": ["quality", "punctuality", "communication"],
  "concerns": ["concern1", "concern2"]
}

Be objective. Neutral is valid. Do not rate based on star rating, analyze the text.`;

  const message = await client.messages.create({
    model: "claude-opus-4-1-20250805",
    max_tokens: 500,
    messages: [
      {
        role: "user",
        content: prompt,
      },
    ],
  });

  const responseText =
    message.content[0].type === "text" ? message.content[0].text : "{}";

  try {
    const analysis = JSON.parse(responseText);

    return {
      sentiment: analysis.sentiment || "neutral",
      confidence: Math.min(Math.max(analysis.confidence || 0.5, 0), 1),
      keywords: Array.isArray(analysis.keywords) ? analysis.keywords : [],
      topics: Array.isArray(analysis.topics) ? analysis.topics : [],
      concerns: Array.isArray(analysis.concerns) ? analysis.concerns : [],
    };
  } catch (error) {
    console.error("Failed to parse sentiment analysis:", error);
    return {
      sentiment: "neutral",
      confidence: 0.5,
      keywords: [],
      topics: [],
      concerns: [],
    };
  }
}

export async function detectSpamIndicators(
  reviewText: string
): Promise<string[]> {
  const indicators: string[] = [];

  // Check for contact information
  if (
    /(\+?1?\d{9,15}|[\w.-]+@[\w.-]+\.\w+|www\.\S+|telegram|whatsapp)/i.test(
      reviewText
    )
  ) {
    indicators.push("contains_contact_info");
  }

  // Check for promotional language
  if (
    /(buy now|click here|limited time|special offer|discount|visit|link|coupon)/i.test(
      reviewText
    )
  ) {
    indicators.push("promotional_language");
  }

  // Check for repeated characters
  if (/(.)\1{4,}/.test(reviewText)) {
    indicators.push("repeated_characters");
  }

  // Check for excessive punctuation
  const punctuationCount = (reviewText.match(/[!?]{2,}/g) || []).length;
  if (punctuationCount > 3) {
    indicators.push("excessive_punctuation");
  }

  // Check for link density
  const links = (reviewText.match(/https?:\/\/\S+/g) || []).length;
  if (links > 0) {
    indicators.push("contains_links");
  }

  // Check for copy-paste detection
  if (reviewText.length > 500 && reviewText.split("\n").length > 10) {
    indicators.push("potentially_copied");
  }

  return indicators;
}

  value_rating: z.number().min(1).max(5),
  professionalism_rating: z.number().min(1).max(5),

  review_images: z.array(z.string().url()).max(5).optional(),

export async function POST(request: NextRequest) {
  const startTime = Date.now();

  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();

    if (authError || !user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    // Parse and validate
    const body = await request.json();
    const data = createReviewSchema.parse(body);

    // Verify booking exists and is completed
    const { data: booking, error: bookingError } = await supabase
      .from("bookings")
      .select("id, provider_id, user_id, status")
      .eq("id", data.booking_id)
      .single();

    if (bookingError || !booking) {
      return NextResponse.json(
        { error: "Booking not found" },
        { status: 404 }
      );
    }

    if (booking.user_id !== user.id) {
      return NextResponse.json(
        { error: "You can only review your own bookings" },
        { status: 403 }
      );
    }

    if (booking.status !== "completed") {
      return NextResponse.json(
        { error: "Booking must be completed before review" },
        { status: 400 }
      );
    }

    // Check if review already exists
    const { data: existingReview } = await supabase
      .from("reviews")
      .select("id")
      .eq("booking_id", data.booking_id)
      .single();

    if (existingReview) {
      return NextResponse.json(
        { error: "Review already exists for this booking" },
        { status: 400 }
      );
    }

    // Perform sentiment analysis (async)
    const sentimentPromise = analyzeReviewSentiment(
      data.review_text,
      data.review_title
    ).catch(() => ({
      sentiment: "neutral" as const,
      confidence: 0.5,
      keywords: [],
      topics: [],
      concerns: [],
    }));

    // Detect spam indicators
    const spamIndicators = await detectSpamIndicators(data.review_text);

    // Wait for sentiment analysis
    const sentimentAnalysis = await sentimentPromise;

    // Create review (triggers will calculate authenticity)
    const { data: review, error: reviewError } = await supabase
      .from("reviews")
      .insert({
        booking_id: data.booking_id,
        provider_id: booking.provider_id,
        user_id: user.id,
        overall_rating: data.overall_rating,
        punctuality_rating: data.punctuality_rating,
        quality_rating: data.quality_rating,
        value_rating: data.value_rating,
        professionalism_rating: data.professionalism_rating,
        review_title: data.review_title,
        review_text: data.review_text,
        review_image_urls: data.review_images || [],
        detected_sentiment: sentimentAnalysis.sentiment,
        sentiment_confidence: sentimentAnalysis.confidence,
        spam_indicators: spamIndicators,
        moderation_status: "pending",
      })
      .select()
      .single();

    if (reviewError) {
      console.error("Review creation error:", reviewError);
      return NextResponse.json(
        { error: "Failed to create review" },
        { status: 500 }
      );
    }

    // Create moderation queue entry (priority calculated by trigger)
    await supabase.from("moderation_queue").insert({
      review_id: review.id,
      priority_score: review.moderation_queue_priority,
      priority_reason: spamIndicators.length > 0 ? "spam_indicators_detected" : "routine_moderation",
      due_at: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString(), // 24h SLA
    });

    // Queue notification email to provider
    await emailQueue.add(
      "review-submitted",
      {
        type: "review_submitted",
        data: {
          provider_id: booking.provider_id,
          review_id: review.id,
          rating: data.overall_rating,
          title: data.review_title,
        },
      },
      defaultJobOptions
    );

    // Async: refresh materialized view
    supabase.rpc("refresh_provider_rating_stats").catch(console.error);

    // Log analytics
    await supabase.from("review_analytics").insert({
      review_id: review.id,
      submitted_at: new Date().toISOString(),
      processing_time_ms: Date.now() - startTime,
      sentiment_detected: sentimentAnalysis.sentiment,
      spam_indicators_found: spamIndicators.length > 0,
    }).catch(console.error);

    return NextResponse.json({
      success: true,
      review: {
        id: review.id,
        status: review.moderation_status,
        submitted_at: review.created_at,
      },
      processing_time_ms: Date.now() - startTime,
    });

  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: "Invalid review data", details: error.errors },
        { status: 400 }
      );
    }

    console.error("Review creation error:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}

typescriptimport { NextRequest, NextResponse } from "next/server";
import { createClient } from "@/lib/supabase/server";
import { emailQueue, defaultJobOptions } from "@/lib/queue/config";
import { z } from "zod";

const moderationSchema = z.object({
  action: z.enum(["approve", "reject", "flag_for_review"]),
  rejection_reason: z.string().optional(),
  moderator_notes: z.string().max(500).optional(),

export async function PATCH(
  request: NextRequest,
  { params }: { params: { id: string } }
) {
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();

    if (authError || !user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    // Verify user is moderator
    const { data: profile } = await supabase
      .from("profiles")
      .select("user_role")
      .eq("id", user.id)
      .single();

    if (profile?.user_role !== "moderator") {
      return NextResponse.json(
        { error: "Only moderators can moderate reviews" },
        { status: 403 }
      );
    }

    // Parse body
    const body = await request.json();
    const { action, rejection_reason, moderator_notes } = moderationSchema.parse(body);

    // Get review and booking info
    const { data: review, error: reviewError } = await supabase
      .from("reviews")
      .select("*, booking:bookings!inner(id, provider_id, user_id)")
      .eq("id", params.id)
      .single();

    if (reviewError || !review) {
      return NextResponse.json(
        { error: "Review not found" },
        { status: 404 }
      );
    }

    if (review.moderation_status !== "pending") {
      return NextResponse.json(
        { error: "Review already moderated" },
        { status: 400 }
      );
    }

    let newStatus: string;
    switch (action) {
      case "approve":
        newStatus = "approved";
        break;
      case "reject":
        if (!rejection_reason) {
          return NextResponse.json(
            { error: "Rejection reason required" },
            { status: 400 }
          );
        }
        newStatus = "rejected";
        break;
      case "flag_for_review":
        newStatus = "flagged_for_review";
        break;
    }

    // Update review
    const { error: updateError } = await supabase
      .from("reviews")
      .update({
        moderation_status: newStatus,
        moderated_at: new Date().toISOString(),
        moderated_by: user.id,
        moderation_notes: moderator_notes,
        rejection_reason: action === "reject" ? rejection_reason : null,
      })
      .eq("id", params.id);

    if (updateError) {
      console.error("Moderation error:", updateError);
      return NextResponse.json(
        { error: "Failed to moderate review" },
        { status: 500 }
      );
    }

    // Update moderation queue
    await supabase
      .from("moderation_queue")
      .update({
        status: "completed",
        completed_at: new Date().toISOString(),
        resolution_time_seconds: Math.floor(
          (Date.now() - new Date(review.created_at).getTime()) / 1000
        ),
      })
      .eq("review_id", params.id);

    // Send notification to reviewer if rejected
    if (action === "reject") {
      const { data: reviewer } = await supabase
        .from("auth.users")
        .select("email")
        .eq("id", review.user_id)
        .single();

      if (reviewer?.email) {
        await emailQueue.add(
          "review-rejected",
          {
            type: "review_rejected",
            data: {
              to: reviewer.email,
              reason: rejection_reason,
              review_id: params.id,
            },
          },
          defaultJobOptions
        );
      }
    }

    // Send notification to provider if approved
    if (action === "approve") {
      const { data: provider } = await supabase
        .from("auth.users")
        .select("email")
        .eq("id", review.booking.provider_id)
        .single();

      if (provider?.email) {
        await emailQueue.add(
          "review-approved",
          {
            type: "review_approved",
            data: {
              to: provider.email,
              review_id: params.id,
              rating: review.overall_rating,
            },
          },
          defaultJobOptions
        );
      }
    }

    // Refresh ratings if approved
    if (action === "approve") {
      await supabase.rpc("refresh_provider_rating_stats").catch(console.error);
    }

    return NextResponse.json({
      success: true,
      review_id: params.id,
      new_status: newStatus,
    });

  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: "Invalid moderation data", details: error.errors },
        { status: 400 }
      );
    }

    console.error("Moderation error:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}

typescriptimport { NextRequest, NextResponse } from "next/server";
import { createClient } from "@/lib/supabase/server";
import { Redis } from "@upstash/redis";
import { z } from "zod";

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!,
});

const listSchema = z.object({
  provider_id: z.string().uuid(),
  sort_by: z.enum(["helpful", "recent", "relevant"]).default("recent"),
  search_query: z.string().optional(),
  min_rating: z.number().min(1).max(5).optional(),
  max_rating: z.number().min(1).max(5).optional(),
  page: z.number().min(1).default(1),
  per_page: z.number().min(1).max(50).default(20),
});

export async function GET(request: NextRequest) {
  const startTime = Date.now();

  try {
    const { searchParams } = new URL(request.url);
    const params = listSchema.parse(Object.fromEntries(searchParams));

    // Check cache
    const cacheKey = `reviews:list:${JSON.stringify(params)}`;
    const cached = await redis.get(cacheKey);

    if (cached) {
      return NextResponse.json({
        ...cached,
        cached: true,
        query_time_ms: Date.now() - startTime,
      });
    }

    const supabase = await createClient();

    // Get total count for pagination
    let countQuery = supabase
      .from("reviews")
      .select("id", { count: "exact", head: true })
      .eq("provider_id", params.provider_id)
      .eq("moderation_status", "approved")
      .is("deleted_at", null);

    if (params.min_rating) {
      countQuery = countQuery.gte("overall_rating", params.min_rating);
    }
    if (params.max_rating) {
      countQuery = countQuery.lte("overall_rating", params.max_rating);
    }

    const { count } = await countQuery;

    // Get reviews using search function
    const { data: reviews, error: searchError } = await supabase.rpc(
      "search_reviews",
      {
        p_provider_id: params.provider_id,
        p_search_query: params.search_query || null,
        p_min_rating: params.min_rating || null,
        p_max_rating: params.max_rating || null,
        p_sort_by: params.sort_by,
        p_limit: params.per_page,
        p_offset: (params.page - 1) * params.per_page,
      }
    );

    if (searchError) {
      throw searchError;
    }

    // Get rating stats
    const { data: stats } = await supabase
      .from("provider_rating_stats")
      .select("*")
      .eq("provider_id", params.provider_id)
      .single();

    const response = {
      reviews: reviews || [],
      pagination: {
        page: params.page,
        per_page: params.per_page,
        total: count || 0,
        total_pages: Math.ceil((count || 0) / params.per_page),
      },
      stats: stats ? {
        average_rating: stats.average_rating,
        total_reviews: stats.total_reviews,
        response_rate: stats.response_rate,
        distribution: {
          five_star: stats.count_5_star,
          four_star: stats.count_4_star,
          three_star: stats.count_3_star,
          two_star: stats.count_2_star,
          one_star: stats.count_1_star,
        },
      } : null,
      query_time_ms: Date.now() - startTime,
    };

    // Cache for 5 minutes
    await redis.setex(cacheKey, 300, response);

    return NextResponse.json(response);

  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: "Invalid parameters", details: error.errors },
        { status: 400 }
      );
    }

    console.error("Review list error:", error);
    return NextResponse.json(
      { error: "Failed to fetch reviews" },
      { status: 500 }
    );
  }
}

          onChange={(val) => setRatings({ ...ratings, punctuality: val })}
        />
        <StarRating
          label="Quality of Work"
          value={ratings.quality}
          onChange={(val) => setRatings({ ...ratings, quality: val })}
        />
        <StarRating
          label="Value for Money"
          value={ratings.value}
          onChange={(val) => setRatings({ ...ratings, value: val })}
        />
        <StarRating
          label="Professionalism"
          value={ratings.professionalism}
          onChange={(val) => setRatings({ ...ratings, professionalism: val })}
        />
      </div>

      {/* Review Title */}
      <div>
        <Label htmlFor="title">Review Title *</Label>
        <Input
          id="title"
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          placeholder="Summarize your experience..."
          maxLength={100}
          required
        />
        <p className="text-xs text-gray-500 mt-1">
          {title.length}/100 characters
        </p>
      </div>

      {/* Review Text */}
      <div>
        <Label htmlFor="text">Your Review *</Label>
        <Textarea
          id="text"
          value={text}
          onChange={(e) => setText(e.target.value)}
          placeholder="Share details about your experience. What went well? What could be improved?"
          rows={8}
          maxLength={2000}
          required
        />
        <p className="text-xs text-gray-500 mt-1">
          {text.length}/2000 characters
        </p>
      </div>

      {/* Guidelines */}
      <div className="bg-blue-50 border border-blue-200 rounded-lg p-4">
        <p className="text-sm font-medium text-blue-900 mb-2">
          Helpful Review Tips:
        </p>
        <ul className="text-sm text-blue-800 space-y-1 list-disc list-inside">
          <li>Be specific about what you liked or disliked</li>
          <li>Describe your experience in detail</li>
          <li>Be honest and fair in your assessment</li>
          <li>Focus on the service provided</li>
        </ul>
      </div>

      {/* Submit */}
      <Button type="submit" disabled={isSubmitting} className="w-full">
        {isSubmitting ? 'Submitting Review...' : 'Submit Review'}
      </Button>
    </form>
  );
}
