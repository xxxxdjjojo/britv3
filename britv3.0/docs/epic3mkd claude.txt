```sql
  result JSONB;
BEGIN
  SELECT to_jsonb(dd.*) INTO result
  FROM dashboard_data dd
  WHERE dd.user_id = p_user_id;
  
  RETURN result;
END;
$$ LANGUAGE plpgsql STABLE SECURITY DEFINER;

-- Async activity logging
CREATE OR REPLACE FUNCTION log_activity_async(
  p_user_id UUID,
  p_action_type activity_type,
  p_details JSONB DEFAULT '{}'::jsonb,
  p_ip_address INET DEFAULT NULL,
  p_user_agent TEXT DEFAULT NULL,
  p_duration_ms INTEGER DEFAULT NULL
)
RETURNS VOID AS $$
BEGIN
  INSERT INTO activity_log (user_id, action_type, details, ip_address, user_agent, duration_ms)
  VALUES (p_user_id, p_action_type, p_details, p_ip_address, p_user_agent, p_duration_ms);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Update preferences
CREATE OR REPLACE FUNCTION update_user_preferences(
  p_user_id UUID,
  p_preferences JSONB
)
RETURNS VOID AS $$
BEGIN
  IF NOT validate_preferences(p_preferences) THEN
    RAISE EXCEPTION 'Invalid preferences structure';
  END IF;
  
  UPDATE profiles
  SET preferences = p_preferences,
      updated_at = NOW()
  WHERE id = p_user_id;
  
  PERFORM log_activity_async(p_user_id, 'profile_updated', 
    jsonb_build_object('updated_field', 'preferences'));
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- ============================================================================
-- TRIGGERS
-- ============================================================================

-- Prevent self role change
CREATE OR REPLACE FUNCTION prevent_self_role_change()
RETURNS TRIGGER AS $$
BEGIN
  IF OLD.user_role IS DISTINCT FROM NEW.user_role THEN
    RAISE EXCEPTION 'Cannot change user role via profile update';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER prevent_role_change
  BEFORE UPDATE ON profiles
  FOR EACH ROW
  WHEN (OLD.user_role IS DISTINCT FROM NEW.user_role)
  EXECUTE FUNCTION prevent_self_role_change();
```

```typescript
import DOMPurify from 'isomorphic-dompurify';

export function sanitizeUserInput(input: string, maxLength: number = 5000): string {
  let sanitized = input.trim().substring(0, maxLength);
  
  sanitized = DOMPurify.sanitize(sanitized, {
    ALLOWED_TAGS: [],
    ALLOWED_ATTR: [],
  });
  
  sanitized = sanitized
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#x27;')
    .replace(/\//g, '&#x2F;');
  
  return sanitized;
}

export function validatePreferences(prefs: any): boolean {
  if (typeof prefs !== 'object' || !prefs.notifications || !prefs.dashboard || !prefs.privacy) {
    return false;
  }
  return true;
}
```

#### Step 2.2: Profile Picture Security

**File:** `src/lib/profile/security.ts`

```typescript
import { createClient } from '@/lib/supabase/server';
import sharp from 'sharp';
import crypto from 'crypto';

interface ProfilePictureConfig {
  maxFileSize: number;
  allowedMimeTypes: string[];
  dimensions: { width: number; height: number };
}

const PROFILE_PICTURE_CONFIG: ProfilePictureConfig = {
  maxFileSize: 5 * 1024 * 1024, // 5MB
  allowedMimeTypes: ['image/jpeg', 'image/png', 'image/webp'],
  dimensions: { width: 400, height: 400 },
};

export async function secureProfilePictureUpload(
  file: File,
  userId: string
): Promise<{ success: boolean; url?: string; error?: string }> {
  
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  
  if (!user || user.id !== userId) {
    return { success: false, error: 'Unauthorized' };
  }
  
  if (file.size > PROFILE_PICTURE_CONFIG.maxFileSize) {
    return { success: false, error: 'File too large. Max: 5MB' };
  }
  
  const arrayBuffer = await file.arrayBuffer();
  const buffer = Buffer.from(arrayBuffer);
  const { fileTypeFromBuffer } = await import('file-type');
  const detectedType = await fileTypeFromBuffer(buffer);
  
  if (!detectedType || !PROFILE_PICTURE_CONFIG.allowedMimeTypes.includes(detectedType.mime)) {
    return { success: false, error: 'Invalid file type' };
  }
  
  const optimizedBuffer = await sharp(buffer)
    .resize(400, 400, { fit: 'cover', position: 'center' })
    .withMetadata({ exif: {} })
    .webp({ quality: 85 })
    .toBuffer();
  
  const hash = crypto.createHash('sha256').update(optimizedBuffer).digest('hex');
  const filename = `avatars/${userId}/${hash}.webp`;
  
  const { data: oldProfile } = await supabase
    .from('profiles')
    .select('avatar_url')
    .eq('id', userId)
    .single();
  
  if (oldProfile?.avatar_url) {
    const oldPath = oldProfile.avatar_url.split('/').slice(-2).join('/');
    await supabase.storage.from('profile-images').remove([oldPath]);
  }
  
  const { data, error } = await supabase.storage
    .from('profile-images')
    .upload(filename, optimizedBuffer, {
      contentType: 'image/webp',
      cacheControl: '31536000',
      upsert: true,
    });
  
  if (error) {
    return { success: false, error: 'Upload failed' };
  }
  
  const { data: urlData } = supabase.storage
    .from('profile-images')
    .getPublicUrl(filename);
  
  await supabase
    .from('profiles')
    .update({ avatar_url: urlData.publicUrl })
    .eq('id', userId);
  
  await supabase.from('activity_log').insert({
    user_id: userId,
    action_type: 'profile_picture_updated',
    details: { filename: file.name },
  });
  
  return { success: true, url: urlData.publicUrl };
}
```

```typescript
import { createClient } from '@/lib/supabase/server';
import { NextRequest, NextResponse } from 'next/server';
import { Redis } from '@upstash/redis';

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!,
});

export async function GET(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    // Check cache
    const cacheKey = `dashboard:${user.id}`;
    const cached = await redis.get(cacheKey);
    
    if (cached) {
      return NextResponse.json({
        ...cached,
        cached: true,
        query_time_ms: Date.now() - startTime,
      });
    }
    
    // Single aggregated query
    const { data: dashboardData, error } = await supabase
      .rpc('get_user_dashboard', { p_user_id: user.id });
    
    if (error) throw error;
    
    const role = dashboardData.user_role;
    let roleSpecificData = {};
    
    // Fetch role-specific data
    if (role === 'homebuyer' || role === 'renter') {
      const { data: savedProps } = await supabase
        .from('saved_properties')
        .select(`
          id,
          created_at,
          listing:listings!inner (
            id,
            price,
            slug,
            property:properties!inner (
              title,
              bedrooms,
              city,
              coordinates
            ),
            primary_image:property_media!inner (
              cdn_url
            )
          )
        `)
        .order('created_at', { ascending: false })
        .limit(5);
      
      const { data: recentSearches } = await supabase
        .from('saved_searches')
        .select('id, name, filters, created_at')
        .order('created_at', { ascending: false })
        .limit(5);
      
      roleSpecificData = {
        saved_properties: savedProps || [],
        recent_searches: recentSearches || [],
      };
      
    } else if (role === 'seller' || role === 'agent' || role === 'landlord') {
      const { data: activeListings } = await supabase
        .from('listings')
        .select(`
          id,
          price,
          status,
          view_count,
          enquiry_count,
          property:properties!inner (
            title,
            bedrooms,
            city
          )
        `)
        .eq('status', 'active')
        .order('listed_date', { ascending: false })
        .limit(10);
      
      roleSpecificData = {
        active_listings: activeListings || [],
      };
      
    } else if (role === 'service_provider') {
      const { data: providerDetails } = await supabase
        .from('service_provider_details')
        .select('*')
        .eq('user_id', user.id)
        .single();
      
      roleSpecificData = {
        provider_details: providerDetails,
        recent_leads: [],
      };
    }
    
    // Recent activity (limited)
    const { data: recentActivity } = await supabase
      .from('activity_log')
      .select('action_type, created_at, details')
      .eq('user_id', user.id)
      .order('created_at', { ascending: false })
      .limit(10);
    
    const response = {
      user: {
        id: user.id,
        email: user.email,
        ...dashboardData,
      },
      role_data: roleSpecificData,
      recent_activity: recentActivity || [],
      query_time_ms: Date.now() - startTime,
    };
    
    // Cache for 5 minutes
    await redis.setex(cacheKey, 300, response);
    
    return NextResponse.json(response);
    
  } catch (error) {
    console.error('Dashboard error:', error);
    return NextResponse.json(
      { error: 'Failed to load dashboard' },
      { status: 500 }
    );
  }
}
```

```typescript
import { createClient } from '@/lib/supabase/server';
import { NextRequest, NextResponse } from 'next/server';
import { sanitizeUserInput, validatePreferences } from '@/lib/validation/sanitize';
import { invalidateDashboardCache } from '@/lib/cache/dashboard';

export async function GET(request: NextRequest) {
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    const { data: profile, error } = await supabase
      .from('profiles')
      .select('*')
      .eq('id', user.id)
      .single();
    
    if (error) throw error;
    
    return NextResponse.json({ profile });
    
  } catch (error) {
    console.error('Profile fetch error:', error);
    return NextResponse.json({ error: 'Failed to fetch profile' }, { status: 500 });
  }
}

export async function PATCH(request: NextRequest) {
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    const body = await request.json();
    const updates: any = {};
    
    if (body.display_name) {
      updates.display_name = sanitizeUserInput(body.display_name, 100);
    }
    
    if (body.bio) {
      updates.bio = sanitizeUserInput(body.bio, 500);
    }
    
    if (body.phone) {
      if (!/^\+?[1-9]\d{1,14}$/.test(body.phone)) {
        return NextResponse.json({ error: 'Invalid phone format' }, { status: 400 });
      }
      updates.phone = body.phone;
    }
    
    if (body.preferences) {
      if (!validatePreferences(body.preferences)) {
        return NextResponse.json({ error: 'Invalid preferences structure' }, { status: 400 });
      }
      updates.preferences = body.preferences;
    }
    
    const { data, error } = await supabase
      .from('profiles')
      .update(updates)
      .eq('id', user.id)
      .select()
      .single();
    
    if (error) throw error;
    
    await invalidateDashboardCache(user.id);
    
    await supabase.rpc('log_activity_async', {
      p_user_id: user.id,
      p_action_type: 'profile_updated',
      p_details: { updated_fields: Object.keys(updates) },
    });
    
    return NextResponse.json({ profile: data });
    
  } catch (error) {
    console.error('Profile update error:', error);
    return NextResponse.json({ error: 'Failed to update profile' }, { status: 500 });
  }
}
```

```typescript
    const formData = await request.formData();
    const file = formData.get('file') as File;
    
    if (!file) {
      return NextResponse.json({ error: 'No file provided' }, { status: 400 });
    }
    
    const result = await secureProfilePictureUpload(file, user.id);
    
    if (!result.success) {
      return NextResponse.json({ error: result.error }, { status: 400 });
    }
    
    return NextResponse.json({ 
      success: true,
      url: result.url,
    });
    
  } catch (error) {
    console.error('Profile picture upload error:', error);
    return NextResponse.json({ error: 'Upload failed' }, { status: 500 });
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
```

```typescript
import { useEffect } from 'react';
import { createClient } from '@/lib/supabase/client';
import { RealtimeChannel } from '@supabase/supabase-js';

export function useRealtime(channel: string, onUpdate: (payload: any) => void) {
  const supabase = createClient();
  
  useEffect(() => {
    const subscription: RealtimeChannel = supabase
      .channel(channel)
      .on('broadcast', { event: 'update' }, ({ payload }) => {
        onUpdate(payload);
      })
      .subscribe();
    
    return () => {
      subscription.unsubscribe();
    };
  }, [channel]);
}
```
