import { createClient } from '@/lib/supabase/server';
import crypto from 'crypto';
import sharp from 'sharp'; // Image processing
import { exifr } from 'exifr'; // EXIF reading
import clamav from 'clamav.js'; // Antivirus scanning

interface UploadSecurityConfig {
  maxFileSize: number;
  allowedMimeTypes: string[];
  allowedExtensions: string[];
  virusScanEnabled: boolean;
  stripExif: boolean;
  generateThumbnails: boolean;
}

const IMAGE_UPLOAD_CONFIG: UploadSecurityConfig = {
  maxFileSize: 10 * 1024 * 1024, // 10MB
  allowedMimeTypes: ['image/jpeg', 'image/png', 'image/webp'],
  allowedExtensions: ['.jpg', '.jpeg', '.png', '.webp'],
  virusScanEnabled: true,
  stripExif: true,
  generateThumbnails: true,
};

export async function secureImageUpload(
  file: File,
  listingId: string,
  userId: string
): Promise<{ success: boolean; url?: string; error?: string }> {
  
  // 1. Validate listing ownership
  const supabase = await createClient();
  const { data: listing, error: listingError } = await supabase
    .from('listings')
    .select('user_id')
    .eq('id', listingId)
    .single();
  
  if (listingError || listing.user_id !== userId) {
    return { success: false, error: 'Unauthorized' };
  }
  
  // 2. File size validation
  if (file.size > IMAGE_UPLOAD_CONFIG.maxFileSize) {
    return { 
      success: false, 
      error: `File too large. Max: ${IMAGE_UPLOAD_CONFIG.maxFileSize / 1024 / 1024}MB` 
    };
  }
  
  // 3. MIME type validation (don't trust client)
  const arrayBuffer = await file.arrayBuffer();
  const buffer = Buffer.from(arrayBuffer);
  const detectedType = await detectMimeType(buffer);
  
  if (!IMAGE_UPLOAD_CONFIG.allowedMimeTypes.includes(detectedType)) {
    return { 
      success: false, 
      error: `Invalid file type: ${detectedType}` 
    };
  }
  
  // 4. Virus scanning (ClamAV or cloud service)
  if (IMAGE_UPLOAD_CONFIG.virusScanEnabled) {
    const isSafe = await scanForVirus(buffer);
    if (!isSafe) {
      // Log security incident
      await supabase.from('security_events').insert({
        user_id: userId,
        event_type: 'malware_upload_blocked',
        details: { filename: file.name, size: file.size }
      });
      return { success: false, error: 'File failed security scan' };
    }
  }
  
  // 5. Strip EXIF data (privacy protection)
  let processedBuffer = buffer;
  if (IMAGE_UPLOAD_CONFIG.stripExif) {
    processedBuffer = await sharp(buffer)
      .rotate() // Auto-rotate based on EXIF
      .withMetadata({ 
        exif: {}, // Remove all EXIF
        icc: true, // Keep color profile
      })
      .toBuffer();
  }
  
  // 6. Optimize image (reduce size, convert to WebP)
  const optimizedBuffer = await sharp(processedBuffer)
    .resize(2400, 1800, { 
      fit: 'inside', 
      withoutEnlargement: true 
    })
    .webp({ quality: 85 })
    .toBuffer();
  
  // 7. Generate secure filename
  const hash = crypto.createHash('sha256').update(optimizedBuffer).digest('hex');
  const filename = `${listingId}/${hash}.webp`;
  
  // 8. Upload to Supabase Storage
  const { data, error } = await supabase.storage
    .from('property-images')
    .upload(filename, optimizedBuffer, {
      contentType: 'image/webp',
      cacheControl: '31536000', // 1 year
      upsert: false, // Prevent overwrites
    });
  
  if (error) {
    return { success: false, error: 'Upload failed' };
  }
  
  // 9. Generate signed URL (short-lived for security)
  const { data: signedUrlData } = await supabase.storage
    .from('property-images')
    .createSignedUrl(filename, 31536000); // 1 year
  
  // 10. Store in database
  await supabase.from('property_media').insert({
    listing_id: listingId,
    media_type: 'image',
    url: signedUrlData.signedUrl,
    file_size: optimizedBuffer.length,
    original_filename: file.name,
  });
  
  // 11. Generate thumbnails (async job)
  generateThumbnails(optimizedBuffer, filename).catch(console.error);
  
  return { 
    success: true, 
    url: signedUrlData.signedUrl 
  };
}

// Helper: Detect actual MIME type from file magic bytes
async function detectMimeType(buffer: Buffer): Promise<string> {
  const { fileTypeFromBuffer } = await import('file-type');
  const type = await fileTypeFromBuffer(buffer);
  return type?.mime || 'application/octet-stream';
}

// Helper: Virus scanning
async function scanForVirus(buffer: Buffer): Promise<boolean> {
  // In production: integrate with ClamAV or CloudMersive
  // For MVP: basic signature checks
  try {
    const scanner = await clamav.createScanner(3310, 'clamav-service');
    const result = await scanner.scanBuffer(buffer);
    return result.isClean;
  } catch (error) {
    console.error('Virus scan error:', error);
    return false; // Fail closed
  }
}

-- Generated column for full-text search
CREATE OR REPLACE FUNCTION properties_description_tsv_trigger() RETURNS trigger AS $$
BEGIN
  NEW.description_tsv :=
    setweight(to_tsvector('property_search', COALESCE(NEW.title, '')), 'A') ||
    setweight(to_tsvector('property_search', COALESCE(NEW.description, '')), 'B') ||
    setweight(to_tsvector('property_search', COALESCE(NEW.address_line1, '')), 'C');
  RETURN NEW;
END
$$ LANGUAGE plpgsql;

CREATE TRIGGER properties_tsv_update
  BEFORE INSERT OR UPDATE OF title, description, address_line1
  ON properties
  FOR EACH ROW
  EXECUTE FUNCTION properties_description_tsv_trigger();

-- Geospatial radius search
CREATE OR REPLACE FUNCTION search_listings_by_radius(
  center_lat FLOAT,
  center_lng FLOAT,
  radius_meters INTEGER,
  p_listing_type listing_type DEFAULT NULL,
  p_min_price NUMERIC DEFAULT NULL,
  p_max_price NUMERIC DEFAULT NULL,
  p_min_bedrooms INTEGER DEFAULT NULL,
  p_property_type property_type DEFAULT NULL,
  p_limit INTEGER DEFAULT 50,
  p_offset INTEGER DEFAULT 0
)
RETURNS TABLE (
  listing_id UUID,
  distance_meters FLOAT,
  price NUMERIC,
  bedrooms INTEGER,
  bathrooms NUMERIC,
  title TEXT,
  primary_image TEXT
) AS $$
BEGIN
  RETURN QUERY
  SELECT 
    sl.listing_id,
    ST_Distance(
      sl.coordinates::geography,
      ST_SetSRID(ST_MakePoint(center_lng, center_lat), 4326)::geography
    ) AS distance_meters,
    sl.price,
    sl.bedrooms,
    sl.bathrooms,
    sl.title,
    sl.primary_image_url
  FROM search_listings sl
  WHERE 
    ST_DWithin(
      sl.coordinates::geography,
      ST_SetSRID(ST_MakePoint(center_lng, center_lat), 4326)::geography,
      radius_meters
    )
    AND (p_listing_type IS NULL OR sl.listing_type = p_listing_type)
    AND (p_min_price IS NULL OR sl.price >= p_min_price)
    AND (p_max_price IS NULL OR sl.price <= p_max_price)
    AND (p_min_bedrooms IS NULL OR sl.bedrooms >= p_min_bedrooms)
    AND (p_property_type IS NULL OR sl.property_type = p_property_type)
  ORDER BY distance_meters
  LIMIT p_limit
  OFFSET p_offset;
END;
$$ LANGUAGE plpgsql STABLE;

-- Full-text search with ranking
CREATE OR REPLACE FUNCTION search_listings_by_text(
  search_query TEXT,
  p_listing_type listing_type DEFAULT NULL,
  p_min_price NUMERIC DEFAULT NULL,
  p_max_price NUMERIC DEFAULT NULL,
  p_limit INTEGER DEFAULT 50,
  p_offset INTEGER DEFAULT 0
)
RETURNS TABLE (
  listing_id UUID,
  rank REAL,
  price NUMERIC,
  title TEXT,
  snippet TEXT
) AS $$
BEGIN
  RETURN QUERY
  SELECT 
    sl.listing_id,
    ts_rank(sl.description_tsv, to_tsquery('property_search', search_query)) AS rank,
    sl.price,
    sl.title,
    ts_headline(
      'property_search',
      sl.title || ' ' || (SELECT description FROM properties WHERE id = sl.property_id),
      to_tsquery('property_search', search_query),
      'MaxWords=50, MinWords=25'
    ) AS snippet
  FROM search_listings sl
  WHERE 
    sl.description_tsv @@ to_tsquery('property_search', search_query)
    AND (p_listing_type IS NULL OR sl.listing_type = p_listing_type)
    AND (p_min_price IS NULL OR sl.price >= p_min_price)
    AND (p_max_price IS NULL OR sl.price <= p_max_price)
  ORDER BY rank DESC
  LIMIT p_limit
  OFFSET p_offset;
END;
$$ LANGUAGE plpgsql STABLE;

import { createClient } from '@/lib/supabase/server';
import { NextRequest, NextResponse } from 'next/server';
import { z } from 'zod';
import { Redis } from '@upstash/redis'; // For caching

// Cache layer
const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!,
});

const searchSchema = z.object({
  // Location
  postcode: z.string().optional(),
  city: z.string().optional(),
  lat: z.number().optional(),
  lng: z.number().optional(),
  radius_miles: z.number().min(0).max(50).default(5),
  
  // Listing type
  listing_type: z.enum(['sale', 'rent']).optional(),
  
  // Price
  min_price: z.number().min(0).optional(),
  max_price: z.number().min(0).optional(),
  
  // Property specs
  min_bedrooms: z.number().min(0).max(20).optional(),
  max_bedrooms: z.number().min(0).max(20).optional(),
  min_bathrooms: z.number().min(0).max(10).optional(),
  property_type: z.array(z.string()).optional(),
  
  // Search
  query: z.string().optional(),
  
  // Sorting & Pagination
  sort_by: z.enum(['price_asc', 'price_desc', 'date_desc', 'relevance']).default('date_desc'),
  page: z.number().min(1).default(1),
  per_page: z.number().min(1).max(100).default(20),
});

type SearchParams = z.infer<typeof searchSchema>;

export async function GET(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    // 1. Parse and validate query parameters
    const { searchParams } = new URL(request.url);
    const rawParams = Object.fromEntries(searchParams.entries());
    
    // Convert string numbers to actual numbers
    const parsedParams = {
      ...rawParams,
      lat: rawParams.lat ? parseFloat(rawParams.lat) : undefined,
      lng: rawParams.lng ? parseFloat(rawParams.lng) : undefined,
      radius_miles: rawParams.radius_miles ? parseFloat(rawParams.radius_miles) : 5,
      min_price: rawParams.min_price ? parseFloat(rawParams.min_price) : undefined,
      max_price: rawParams.max_price ? parseFloat(rawParams.max_price) : undefined,
      min_bedrooms: rawParams.min_bedrooms ? parseInt(rawParams.min_bedrooms) : undefined,
      max_bedrooms: rawParams.max_bedrooms ? parseInt(rawParams.max_bedrooms) : undefined,
      min_bathrooms: rawParams.min_bathrooms ? parseFloat(rawParams.min_bathrooms) : undefined,
      property_type: rawParams.property_type ? rawParams.property_type.split(',') : undefined,
      page: rawParams.page ? parseInt(rawParams.page) : 1,
      per_page: rawParams.per_page ? parseInt(rawParams.per_page) : 20,
    };
    
    const params = searchSchema.parse(parsedParams);
    
    // 2. Generate cache key
    const cacheKey = `search:${JSON.stringify(params)}`;
    
    // 3. Check cache (30-second TTL for search results)
    const cached = await redis.get(cacheKey);
    if (cached) {
      return NextResponse.json({
        ...cached,
        cached: true,
        query_time_ms: Date.now() - startTime,
      });
    }
    
    // 4. Execute search query
    const supabase = await createClient();
    
    let query;
    
    // Choose search strategy based on inputs
    if (params.lat && params.lng) {
      // Geospatial search
      query = performGeospatialSearch(supabase, params);
    } else if (params.query) {
      // Full-text search
      query = performTextSearch(supabase, params);
    } else {
      // Standard filter search (uses materialized view)
      query = performFilterSearch(supabase, params);
    }
    
    // Execute query
    const { data: listings, error, count } = await query;
    
    if (error) {
      console.error('Search error:', error);
      return NextResponse.json(
        { error: 'Search failed' },
        { status: 500 }
      );
    }
    
    // 5. Calculate pagination metadata
    const totalPages = Math.ceil((count || 0) / params.per_page);
    
    const response = {
      listings: listings || [],
      pagination: {
        current_page: params.page,
        per_page: params.per_page,
        total_results: count || 0,
        total_pages: totalPages,
        has_next: params.page < totalPages,
        has_prev: params.page > 1,
      },
      filters_applied: params,
      query_time_ms: Date.now() - startTime,
    };
    
    // 6. Cache result (30 seconds)
    await redis.setex(cacheKey, 30, response);
    
    // 7. Log search analytics (async, non-blocking)
    logSearchAnalytics(params, count || 0, Date.now() - startTime).catch(console.error);
    
    return NextResponse.json(response);
    
  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: 'Invalid search parameters', details: error.errors },
        { status: 400 }
      );
    }
    
    console.error('Search error:', error);
    return NextResponse.json(
      { error: 'Internal server error' },
      { status: 500 }
    );
  }
}

// Strategy 1: Geospatial search using PostGIS
function performGeospatialSearch(supabase: any, params: SearchParams) {
  const radiusMeters = (params.radius_miles || 5) * 1609.34; // Convert to meters
  const offset = (params.page - 1) * params.per_page;
  
  return supabase.rpc('search_listings_by_radius', {
    center_lat: params.lat,
    center_lng: params.lng,
    radius_meters: radiusMeters,
    p_listing_type: params.listing_type || null,
    p_min_price: params.min_price || null,
    p_max_price: params.max_price || null,
    p_min_bedrooms: params.min_bedrooms || null,
    p_property_type: params.property_type?.[0] || null,
    p_limit: params.per_page,
    p_offset: offset,
  }, { count: 'exact' });
}

// Strategy 2: Full-text search
function performTextSearch(supabase: any, params: SearchParams) {
  const offset = (params.page - 1) * params.per_page;
  
  // Convert query to tsquery format (handle multi-word)
  const tsquery = params.query!
    .trim()
    .split(/\s+/)
    .map(word => `${word}:*`) // Prefix matching
    .join(' & '); // AND operator
  
  return supabase.rpc('search_listings_by_text', {
    search_query: tsquery,
    p_listing_type: params.listing_type || null,
    p_min_price: params.min_price || null,
    p_max_price: params.max_price || null,
    p_limit: params.per_page,
    p_offset: offset,
  }, { count: 'exact' });
}

// Strategy 3: Filter-based search (materialized view)
function performFilterSearch(supabase: any, params: SearchParams) {
  let query = supabase
    .from('search_listings')
    .select('*', { count: 'exact' });
  
  // Apply filters
  if (params.listing_type) {
    query = query.eq('listing_type', params.listing_type);
  }
  
  if (params.min_price !== undefined) {
    query = query.gte('price', params.min_price);
  }
  
  if (params.max_price !== undefined) {
    query = query.lte('price', params.max_price);
  }
  
  if (params.min_bedrooms !== undefined) {
    query = query.gte('bedrooms', params.min_bedrooms);
  }
  
  if (params.max_bedrooms !== undefined) {
    query = query.lte('bedrooms', params.max_bedrooms);
  }
  
  if (params.property_type && params.property_type.length > 0) {
    query = query.in('property_type', params.property_type);
  }
  
  if (params.city) {
    query = query.ilike('city', `%${params.city}%`);
  }
  
  if (params.postcode) {
    query = query.eq('postcode', params.postcode.toUpperCase());
  }
  
  // Sorting
  switch (params.sort_by) {
    case 'price_asc':
      query = query.order('price', { ascending: true });
      break;
    case 'price_desc':
      query = query.order('price', { ascending: false });
      break;
    case 'date_desc':
      query = query.order('listed_date', { ascending: false });
      break;
    default:
      query = query.order('listed_date', { ascending: false });
  }
  
  // Pagination
  const offset = (params.page - 1) * params.per_page;
  query = query.range(offset, offset + params.per_page - 1);
  
  return query;
}

// Analytics logging (async)
async function logSearchAnalytics(
  params: SearchParams,
  resultCount: number,
  durationMs: number
) {
  const supabase = await createClient();
  
  // Get user ID if authenticated
  const { data: { user } } = await supabase.auth.getUser();
  
  await supabase.from('search_analytics').insert({
    user_id: user?.id || null,
    filters: params,
    result_count: resultCount,
    query_duration_ms: durationMs,
  });
}

import { useEffect, useRef, useState } from 'react';
import mapboxgl from 'mapbox-gl';
import Supercluster from 'supercluster';
import type { BBox, GeoJsonProperties } from 'geojson';

interface Property {
  id: string;
  lat: number;
  lng: number;
  price: number;
  bedrooms: number;
  image_url: string;
  slug: string;
}

interface PropertyMapProps {
  properties: Property[];
  onPropertyClick?: (property: Property) => void;
}

export function PropertyMap({ properties, onPropertyClick }: PropertyMapProps) {
  const mapContainer = useRef<HTMLDivElement>(null);
  const map = useRef<mapboxgl.Map | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  
  const clusterIndex = useRef<Supercluster | null>(null);
  
  useEffect(() => {
    if (!mapContainer.current) return;
    
    // Initialize map
    mapboxgl.accessToken = process.env.NEXT_PUBLIC_MAPBOX_TOKEN!;
    
    map.current = new mapboxgl.Map({
      container: mapContainer.current,
      style: 'mapbox://styles/mapbox/streets-v12',
      center: [-0.1276, 51.5074], // London default
      zoom: 10,
      maxZoom: 18,
    });
    
    map.current.on('load', () => {
      setIsLoading(false);
      initializeClusters();
    });
    
    // Cleanup
    return () => {
      map.current?.remove();
    };
  }, []);
  
  useEffect(() => {
    if (!map.current || !map.current.loaded()) return;
    initializeClusters();
  }, [properties]);
  
  function initializeClusters() {
    if (!map.current) return;
    
    // Convert properties to GeoJSON features
    const points: GeoJSON.Feature<GeoJSON.Point, GeoJsonProperties>[] = properties.map(p => ({
      type: 'Feature',
      properties: {
        id: p.id,
        price: p.price,
        bedrooms: p.bedrooms,
        image_url: p.image_url,
        slug: p.slug,
      },
      geometry: {
        type: 'Point',
        coordinates: [p.lng, p.lat],
      },
    }));
    
    // Initialize Supercluster
    clusterIndex.current = new Supercluster({
      radius: 60, // Cluster radius in pixels
      maxZoom: 16,
      minPoints: 3, // Minimum points to form a cluster
    });
    
    clusterIndex.current.load(points);
    
    // Add sources and layers
    if (map.current.getSource('properties')) {
      (map.current.getSource('properties') as mapboxgl.GeoJSONSource).setData({
        type: 'FeatureCollection',
        features: points,
      });
    } else {
      map.current.addSource('properties', {
        type: 'geojson',
        data: {
          type: 'FeatureCollection',
          features: points,
        },
        cluster: true,
        clusterMaxZoom: 14,
        clusterRadius: 50,
      });
      
      // Cluster circles
      map.current.addLayer({
        id: 'clusters',
        type: 'circle',
        source: 'properties',
        filter: ['has', 'point_count'],
        paint: {
          'circle-color': [
            'step',
            ['get', 'point_count'],
            '#4ECDC4', // 0-10 properties
            10,
            '#7FB685', // 10-30 properties
            30,
            '#FF6B35', // 30+ properties
          ],
          'circle-radius': [
            'step',
            ['get', 'point_count'],
            20, // 0-10 properties
            10,
            30, // 10-30 properties
            30,
            40, // 30+ properties
          ],
        },
      });
      
      // Cluster count labels
      map.current.addLayer({
        id: 'cluster-count',
        type: 'symbol',
        source: 'properties',
        filter: ['has', 'point_count'],
        layout: {
          'text-field': '{point_count_abbreviated}',
          'text-font': ['DIN Offc Pro Medium', 'Arial Unicode MS Bold'],
          'text-size': 12,
        },
        paint: {
          'text-color': '#ffffff',
        },
      });
      
      // Individual property markers
      map.current.addLayer({
        id: 'unclustered-point',
        type: 'circle',
        source: 'properties',
        filter: ['!', ['has', 'point_count']],
        paint: {
          'circle-color': '#005F73',
          'circle-radius': 8,
          'circle-stroke-width': 2,
          'circle-stroke-color': '#ffffff',
        },
      });
      
      // Click handlers
      map.current.on('click', 'clusters', (e) => {
        if (!map.current || !e.features?.[0]) return;
        
        const features = map.current.queryRenderedFeatures(e.point, {
          layers: ['clusters'],
        });
        const clusterId = features[0].properties?.cluster_id;
        
        if (clusterIndex.current) {
          const zoom = clusterIndex.current.getClusterExpansionZoom(clusterId);
          map.current.easeTo({
            center: (features[0].geometry as GeoJSON.Point).coordinates as [number, number],
            zoom: zoom,
          });
        }
      });
      
      map.current.on('click', 'unclustered-point', (e) => {
        if (!e.features?.[0]) return;
        
        const props = e.features[0].properties;
        if (props && onPropertyClick) {
          onPropertyClick({
            id: props.id,
            lat: (e.features[0].geometry as GeoJSON.Point).coordinates[1],
            lng: (e.features[0].geometry as GeoJSON.Point).coordinates[0],
            price: props.price,
            bedrooms: props.bedrooms,
            image_url: props.image_url,
            slug: props.slug,
          });
        }
        
        // Show popup
        new mapboxgl.Popup()
          .setLngLat((e.features[0].geometry as GeoJSON.Point).coordinates as [number, number])
          .setHTML(`
            <div class="p-2">
              <img src="${props?.image_url}" alt="Property" class="w-48 h-32 object-cover mb-2 rounded" />
              <p class="font-bold text-lg">£${props?.price?.toLocaleString()}</p>
              <p class="text-sm">${props?.bedrooms} bedrooms</p>
              <a href="/property/${props?.slug}" class="text-blue-600 text-sm">View details</a>
            </div>
          `)
          .addTo(map.current);
      });
      
      // Cursor styles
      map.current.on('mouseenter', 'clusters', () => {
        if (map.current) map.current.getCanvas().style.cursor = 'pointer';
      });
      map.current.on('mouseleave', 'clusters', () => {
        if (map.current) map.current.getCanvas().style.cursor = '';
      });
      map.current.on('mouseenter', 'unclustered-point', () => {
        if (map.current) map.current.getCanvas().style.cursor = 'pointer';
      });
      map.current.on('mouseleave', 'unclustered-point', () => {
        if (map.current) map.current.getCanvas().style.cursor = '';
      });
    }
    
    // Fit bounds to show all properties
    if (properties.length > 0) {
      const bounds = new mapboxgl.LngLatBounds();
      properties.forEach(p => bounds.extend([p.lng, p.lat]));
      map.current.fitBounds(bounds, { padding: 50 });
    }
  }
  
  return (
    <div className="relative w-full h-full">
      <div ref={mapContainer} className="w-full h-full" />
      {isLoading && (
        <div className="absolute inset-0 flex items-center justify-center bg-gray-100">
          <p>Loading map...</p>
        </div>
      )}
    </div>
  );
}

typescript// lib/image/optimizer.ts
import sharp from 'sharp';

export async function optimizeImage(buffer: Buffer): Promise<{
  webp: Buffer;
  thumbnail: Buffer;
  dimensions: { width: number; height: number };
}> {
  const metadata = await sharp(buffer).metadata();
  
  const webp = await sharp(buffer)
    .resize(2400, 1800, { fit: 'inside', withoutEnlargement: true })
    .webp({ quality: 85, effort: 6 })
    .toBuffer();
  
  const thumbnail = await sharp(buffer)
    .resize(400, 300, { fit: 'cover' })
    .webp({ quality: 75 })
    .toBuffer();
  
  return {
    webp,
    thumbnail,
    dimensions: {
      width: metadata.width || 0,
      height: metadata.height || 0,
    },
  };
}

export function PropertySearch() {
  const [filters, setFilters] = useState({
    listing_type: 'sale',
    min_price: 0,
    max_price: 1000000,
    min_bedrooms: 1,
  });
  
  const [results, setResults] = useState([]);
  const [isLoading, setIsLoading] = useState(false);
  const [page, setPage] = useState(1);
  
  // Debounce search to avoid excessive API calls
  const debouncedFilters = useDebounce(filters, 500);
  
  useEffect(() => {
    performSearch();
  }, [debouncedFilters, page]);
  
  async function performSearch() {
    setIsLoading(true);
    
    const params = new URLSearchParams({
      ...debouncedFilters,
      page: page.toString(),
      per_page: '20',
    });
    
    try {
      const response = await fetch(`/api/search?${params}`);
      const data = await response.json();
      setResults(data.listings);
    } catch (error) {
      console.error('Search failed:', error);
    } finally {
      setIsLoading(false);
    }
  }
  
  return (
    <div className="grid grid-cols-4 gap-6">
      <aside className="col-span-1">
        <SearchFilters filters={filters} onChange={setFilters} />
      </aside>
      
      <main className="col-span-3">
        <SearchResults
          results={results}
          isLoading={isLoading}
          page={page}
          onPageChange={setPage}
        />
      </main>
    </div>
  );

'use client';

import { useState, useEffect } from 'react';
import { createClient } from '@/lib/supabase/client';

export function SavedProperties() {
  const [saved, setSaved] = useState([]);
  const supabase = createClient();
  
  useEffect(() => {
    loadSavedProperties();
  }, []);
  
  async function loadSavedProperties() {
    const { data } = await supabase
      .from('saved_properties')
      .select(`
        id,
        notes,
        created_at,
        listing:listings (
          id,
          price,
          slug,
          property:properties (
            title,
            bedrooms,
            city
          )
        )
      `)
      .order('created_at', { ascending: false });
    
    setSaved(data || []);
  }
  
  async function toggleSave(listingId: string) {
    const existing = saved.find(s => s.listing.id === listingId);
    
    if (existing) {
      await supabase.from('saved_properties').delete().eq('id', existing.id);
    } else {
      await supabase.from('saved_properties').insert({ listing_id: listingId });
    }
    
    loadSavedProperties();
  }
  
  return (
    <div>
      <h2 className="text-2xl font-bold mb-4">Saved Properties</h2>
      <div className="grid grid-cols-3 gap-4">
        {saved.map((item) => (
          <PropertyCard
            key={item.id}
            listing={item.listing}
            isSaved={true}
            onToggleSave={() => toggleSave(item.listing.id)}
          />
        ))}
      </div>
    </div>
  );
}
