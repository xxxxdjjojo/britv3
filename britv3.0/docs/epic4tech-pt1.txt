-- ============================================================================
-- MATERIALIZED VIEW FOR PROVIDER MARKETPLACE
-- ============================================================================
CREATE MATERIALIZED VIEW provider_marketplace AS
SELECT 
  spd.user_id AS provider_id,
  spd.business_name,
  spd.business_description,
  spd.services,
  spd.service_postcodes,
  spd.base_location,
  spd.pricing,
  spd.years_in_business,
  spd.completed_jobs_count,
  spd.average_response_time_hours,
  spd.slug,
  
  p.provider_verification_status,
  p.provider_verified_at,
  p.avatar_url,
  
  -- Aggregated review stats
  COALESCE(AVG(r.overall_rating), 0) AS average_rating,
  COUNT(DISTINCT r.id) AS review_count,
  
  -- Recent activity indicator
  MAX(b.updated_at) AS last_job_date,
  
  -- Search vector for full-text search
  to_tsvector('english',
    COALESCE(spd.business_name, '') || ' ' ||
    COALESCE(spd.business_description, '') || ' ' ||
    COALESCE(array_to_string(spd.services::TEXT[], ' '), '')
  ) AS search_vector

FROM service_provider_details spd
JOIN profiles p ON spd.user_id = p.id
LEFT JOIN reviews r ON spd.user_id = r.provider_id 
  AND r.deleted_at IS NULL 
  AND r.moderation_status = 'approved'
LEFT JOIN bookings b ON spd.user_id = b.provider_id
WHERE p.provider_verification_status = 'verified'
  AND p.deleted_at IS NULL
GROUP BY 
  spd.user_id,
  spd.business_name,
  spd.business_description,
  spd.services,
  spd.service_postcodes,
  spd.base_location,
  spd.pricing,
  spd.years_in_business,
  spd.completed_jobs_count,
  spd.average_response_time_hours,
  spd.slug,
  p.provider_verification_status,
  p.provider_verified_at,
  p.avatar_url;

-- Search providers by location and service
CREATE OR REPLACE FUNCTION search_providers(
  p_service_category service_category DEFAULT NULL,
  p_postcode TEXT DEFAULT NULL,
  p_lat FLOAT DEFAULT NULL,
  p_lng FLOAT DEFAULT NULL,
  p_radius_miles INTEGER DEFAULT 25,
  p_min_rating NUMERIC DEFAULT NULL,
  p_search_query TEXT DEFAULT NULL,
  p_limit INTEGER DEFAULT 20,
  p_offset INTEGER DEFAULT 0
)
RETURNS TABLE (
  provider_id UUID,
  business_name TEXT,
  services service_category[],
  average_rating NUMERIC,
  review_count BIGINT,
  distance_miles NUMERIC,
  slug TEXT
) AS $$
DECLARE
  search_location GEOGRAPHY;
  radius_meters INTEGER;
BEGIN
  -- Convert radius to meters
  radius_meters := p_radius_miles * 1609;
  
  -- Create search location if coordinates provided
  IF p_lat IS NOT NULL AND p_lng IS NOT NULL THEN
    search_location := ST_SetSRID(ST_MakePoint(p_lng, p_lat), 4326)::geography;
  END IF;
  
  RETURN QUERY
  SELECT 
    pm.provider_id,
    pm.business_name,
    pm.services,
    ROUND(pm.average_rating, 2) AS average_rating,
    pm.review_count,
    CASE 
      WHEN search_location IS NOT NULL AND pm.base_location IS NOT NULL THEN
        ROUND((ST_Distance(pm.base_location, search_location) / 1609)::NUMERIC, 1)
      ELSE NULL
    END AS distance_miles,
    pm.slug
  FROM provider_marketplace pm
  WHERE 
    -- Service category filter
    (p_service_category IS NULL OR p_service_category = ANY(pm.services))
    
    -- Location filter
    AND (
      search_location IS NULL 
      OR pm.base_location IS NULL
      OR ST_DWithin(pm.base_location, search_location, radius_meters)
      OR p_postcode = ANY(pm.service_postcodes)
    )
    
    -- Rating filter
    AND (p_min_rating IS NULL OR pm.average_rating >= p_min_rating)
    
    -- Text search filter
    AND (
      p_search_query IS NULL 
      OR pm.search_vector @@ plainto_tsquery('english', p_search_query)
    )
  ORDER BY
    -- Prioritize exact postcode matches
    CASE WHEN p_postcode = ANY(pm.service_postcodes) THEN 0 ELSE 1 END,
    -- Then by distance
    CASE WHEN search_location IS NOT NULL THEN ST_Distance(pm.base_location, search_location) ELSE 999999999 END,
    -- Then by rating
    pm.average_rating DESC,
    -- Finally by review count
    pm.review_count DESC
  LIMIT p_limit
  OFFSET p_offset;
END;
$$ LANGUAGE plpgsql STABLE;

-- Get matching providers for RFQ
CREATE OR REPLACE FUNCTION get_matching_providers_for_rfq(rfq_id UUID)
RETURNS TABLE (
  provider_id UUID,
  business_name TEXT,
  email TEXT,
  match_score INTEGER
) AS $$
DECLARE
  rfq_record RECORD;
BEGIN
  -- Get RFQ details
  SELECT sr.service_category, sr.property_location, sr.property_postcode
  INTO rfq_record
  FROM service_requests sr
  WHERE sr.id = rfq_id;
  
  IF NOT FOUND THEN
    RAISE EXCEPTION 'RFQ not found: %', rfq_id;
  END IF;
  
  RETURN QUERY
  SELECT 
    pm.provider_id,
    pm.business_name,
    au.email,
    (
      -- Score based on service match (50 points)
      CASE WHEN rfq_record.service_category = ANY(pm.services) THEN 50 ELSE 0 END +
      
      -- Score based on postcode match (30 points)
      CASE WHEN rfq_record.property_postcode = ANY(pm.service_postcodes) THEN 30 ELSE 0 END +
      
      -- Score based on location proximity (20 points max)
      CASE 
        WHEN pm.base_location IS NOT NULL AND rfq_record.property_location IS NOT NULL THEN
          GREATEST(0, 20 - (ST_Distance(pm.base_location, rfq_record.property_location) / 1609)::INTEGER)
        ELSE 0
      END +
      
      -- Bonus for high ratings (10 points max)
      LEAST(10, (pm.average_rating * 2)::INTEGER)
    ) AS match_score
  FROM provider_marketplace pm
  JOIN auth.users au ON pm.provider_id = au.id
  WHERE 
    -- Must offer the requested service
    rfq_record.service_category = ANY(pm.services)
    
    -- Must be verified
    AND pm.provider_verification_status = 'verified'
    
    -- Check if within service area (postcode OR reasonable distance)
    AND (
      rfq_record.property_postcode = ANY(pm.service_postcodes)
      OR (
        pm.base_location IS NOT NULL 
        AND rfq_record.property_location IS NOT NULL
        AND ST_DWithin(pm.base_location, rfq_record.property_location, 80000) -- 50 miles
      )
    )
  ORDER BY match_score DESC
  LIMIT 20; -- Max 20 providers per RFQ
END;
$$ LANGUAGE plpgsql STABLE;

  credentials: {

interface UploadResult {
  url: string;
  key: string;
  size: number;
}

export async function uploadProviderDocument(
  file: Buffer,
  providerId: string,
  documentType: string,
  mimeType: string
): Promise<UploadResult> {
  
  // Generate secure filename
  const hash = crypto.createHash('sha256').update(file).digest('hex').substring(0, 16);
  const timestamp = Date.now();
  const ext = mimeType.split('/')[1];
  const key = `providers/${providerId}/documents/${documentType}/${timestamp}-${hash}.${ext}`;
  
  // Encrypt if PDF (for sensitive docs)
  let processedFile = file;
  if (mimeType === 'application/pdf') {
    // TODO: Implement PDF encryption
    processedFile = file; // For now, store as-is
  }
  
  // Optimize images
  if (mimeType.startsWith('image/')) {
    processedFile = await sharp(file)
      .resize(2400, 2400, { fit: 'inside', withoutEnlargement: true })
      .withMetadata({ exif: {} }) // Strip EXIF
      .jpeg({ quality: 85 })
      .toBuffer();
  }
  
  // Upload to R2
  await R2_CLIENT.send(new PutObjectCommand({
    Bucket: BUCKET_NAME,
    Key: key,
    Body: processedFile,
    ContentType: mimeType,
    CacheControl: 'private, max-age=31536000', // Private cache for documents
  }));
  
  return {
    url: `${CDN_URL}/${key}`,
    key,
    size: processedFile.length,
  };
}

export async function deleteProviderDocument(key: string): Promise<void> {
  await R2_CLIENT.send(new DeleteObjectCommand({
    Bucket: BUCKET_NAME,
    Key: key,
  }));
}

export async function getSignedDocumentUrl(key: string, expiresIn: number = 3600): Promise<string> {
  const command = new GetObjectCommand({
    Bucket: BUCKET_NAME,
    Key: key,
  });
  
  return await getSignedUrl(R2_CLIENT, command, { expiresIn });
}

  threats: string[];

export async function scanFileForViruses(fileBuffer: Buffer): Promise<VirusScanResult> {
  try {
    const formData = new FormData();
    formData.append('file', new Blob([fileBuffer]));
    
    const response = await fetch('https://www.virustotal.com/vtapi/v2/file/scan', {
      method: 'POST',
      headers: {
        'x-apikey': process.env.VIRUSTOTAL_API_KEY!,
      },
      body: formData,
    });
    
    if (!response.ok) {
      throw new Error('Virus scan failed');
    }
    
    const data = await response.json();
    
    // Poll for results
    await new Promise(resolve => setTimeout(resolve, 5000)); // Wait 5s
    
    const resultResponse = await fetch(
      `https://www.virustotal.com/vtapi/v2/file/report?apikey=${process.env.VIRUSTOTAL_API_KEY}&resource=${data.resource}`
    );
    
    const result = await resultResponse.json();
    
    return {
      isClean: result.positives === 0,
      threats: result.positives > 0 ? result.scans : [],
      scanDate: new Date(),
    };
  } catch (error) {
    console.error('Virus scan error:', error);
    // Fail closed - mark as potentially infected
    return {
      isClean: false,
      threats: ['Scan failed - manual review required'],
      scanDate: new Date(),
    };
  }
}

typescriptimport { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { uploadProviderDocument } from '@/lib/storage/r2';
import { scanFileForViruses } from '@/lib/security/virus-scan';
import { z } from 'zod';

const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10MB
const ALLOWED_MIME_TYPES = [
  'application/pdf',
  'image/jpeg',
  'image/png',
  'image/webp',
];

const uploadSchema = z.object({
  document_type: z.enum([
    'identity_proof',
    'qualification_certificate',
    'insurance_certificate',
    'business_registration',
    'dbs_check',
    'reference_letter',
  ]),
  document_name: z.string().min(1).max(200),
  expiry_date: z.string().optional(), // ISO date string
  document_data: z.record(z.any()).optional(), // Additional metadata
});

export async function POST(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    // Check if user is a service provider
    const { data: profile } = await supabase
      .from('profiles')
      .select('user_role')
      .eq('id', user.id)
      .single();
    
    if (profile?.user_role !== 'service_provider') {
      return NextResponse.json(
        { error: 'Only service providers can upload documents' },
        { status: 403 }
      );
    }
    
    // Parse multipart form data
    const formData = await request.formData();
    const file = formData.get('file') as File;
    const metadata = JSON.parse(formData.get('metadata') as string);
    
    // Validate metadata
    const validatedMetadata = uploadSchema.parse(metadata);
    
    if (!file) {
      return NextResponse.json({ error: 'No file provided' }, { status: 400 });
    }
    
    // File size validation
    if (file.size > MAX_FILE_SIZE) {
      return NextResponse.json(
        { error: 'File too large. Maximum 10MB' },
        { status: 400 }
      );
    }
    
    // MIME type validation
    if (!ALLOWED_MIME_TYPES.includes(file.type)) {
      return NextResponse.json(
        { error: 'Invalid file type. Only PDF, JPG, PNG, WebP allowed' },
        { status: 400 }
      );
    }
    
    // Convert to buffer
    const arrayBuffer = await file.arrayBuffer();
    const buffer = Buffer.from(arrayBuffer);
    
    // Virus scan (async - don't block upload)
    const virusScan = scanFileForViruses(buffer).catch(err => {
      console.error('Virus scan failed:', err);
      return { isClean: false, threats: ['Scan failed'], scanDate: new Date() };
    });
    
    // Upload to R2
    const uploadResult = await uploadProviderDocument(
      buffer,
      user.id,
      validatedMetadata.document_type,
      file.type
    );
    
    // Wait for virus scan
    const scanResult = await virusScan;
    
    if (!scanResult.isClean) {
      // Delete uploaded file
      await deleteProviderDocument(uploadResult.key);
      
      return NextResponse.json(
        { error: 'File contains malware and was rejected' },
        { status: 400 }
      );
    }
    
    // Save to database
    const { data: document, error: dbError } = await supabase
      .from('provider_documents')
      .insert({
        provider_id: user.id,
        document_type: validatedMetadata.document_type,
        document_name: validatedMetadata.document_name,
        file_url: uploadResult.url,
        file_size_bytes: uploadResult.size,
        file_mime_type: file.type,
        virus_scan_status: 'clean',
        virus_scan_date: scanResult.scanDate.toISOString(),
        document_data: validatedMetadata.document_data || {},
        expiry_date: validatedMetadata.expiry_date || null,
      })
      .select()
      .single();
    
    if (dbError) {
      console.error('Database error:', dbError);
      // Clean up uploaded file
      await deleteProviderDocument(uploadResult.key);
      
      return NextResponse.json(
        { error: 'Failed to save document' },
        { status: 500 }
      );
    }
    
    // Update provider verification status to pending review
    await supabase
      .from('profiles')
      .update({ provider_verification_status: 'pending_review' })
      .eq('id', user.id);
    
    return NextResponse.json({
      success: true,
      document,
      upload_time_ms: Date.now() - startTime,
    });
    
  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: 'Invalid metadata', details: error.errors },
        { status: 400 }
      );
    }
    
    console.error('Document upload error:', error);
    return NextResponse.json(
      { error: 'Internal server error' },
      { status: 500 }
    );
  }
}

import { createClient } from '@/lib/supabase/server';
import { z } from 'zod';

const acceptQuoteSchema = z.object({
  quote_id: z.string().uuid(),
  scheduled_start_date: z.string().optional(), // ISO date
  user_notes: z.string().max(1000).optional(),
});

export async function POST(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    // Parse and validate
    const body = await request.json();
    const data = acceptQuoteSchema.parse(body);
    
    // Get quote with RFQ details
    const { data: quote, error: quoteError } = await supabase
      .from('quotes')
      .select(`
        *,
        service_request:service_requests!inner(
          id,
          user_id,
          title,
          status
        )
      `)
      .eq('id', data.quote_id)
      .single();
    
    if (quoteError || !quote) {
      return NextResponse.json(
        { error: 'Quote not found' },
        { status: 404 }
      );
    }
    
    // Verify user owns the RFQ
    if (quote.service_request.user_id !== user.id) {
      return NextResponse.json(
        { error: 'You can only accept quotes for your own requests' },
        { status: 403 }
      );
    }
    
    // Verify quote is still valid
    if (quote.status !== 'sent' && quote.status !== 'viewed') {
      return NextResponse.json(
        { error: 'This quote can no longer be accepted' },
        { status: 400 }
      );
    }
    
    if (new Date(quote.valid_until) < new Date()) {
      return NextResponse.json(
        { error: 'This quote has expired' },
        { status: 400 }
      );
    }
    
    // Start transaction-like operations
    // 1. Update quote status to accepted
    const { error: quoteUpdateError } = await supabase
      .from('quotes')
      .update({
        status: 'accepted',
        accepted_at: new Date().toISOString(),
      })
      .eq('id', data.quote_id)
      .eq('status', quote.status); // Optimistic locking
    
    if (quoteUpdateError) {
      // Likely concurrent acceptance - check if quote already accepted
      const { data: checkQuote } = await supabase
        .from('quotes')
        .select('status')
        .eq('id', data.quote_id)
        .single();
      
      if (checkQuote?.status === 'accepted') {
        return NextResponse.json(
          { error: 'This quote has already been accepted' },
          { status: 409 }
        );
      }
      
      throw quoteUpdateError;
    }
    
    // 2. Decline other quotes for this RFQ
    await supabase
      .from('quotes')
      .update({ status: 'declined' })
      .eq('service_request_id', quote.service_request_id)
      .neq('id', data.quote_id)
      .in('status', ['sent', 'viewed']);
    
    // 3. Update RFQ status to awarded
    await supabase
      .from('service_requests')
      .update({ status: 'awarded' })
      .eq('id', quote.service_request_id);
    
    // 4. Create booking
    const { data: booking, error: bookingError } = await supabase
      .from('bookings')
      .insert({
        service_request_id: quote.service_request_id,
        quote_id: data.quote_id,
        user_id: user.id,
        provider_id: quote.provider_id,
        scheduled_start_date: data.scheduled_start_date || null,
        estimated_duration_hours: quote.estimated_duration_hours,
        user_notes: data.user_notes || null,
        status: 'pending_confirmation',
      })
      .select()
      .single();
    
    if (bookingError) {
      console.error('Booking creation error:', bookingError);
      
      // Rollback quote status
      await supabase
        .from('quotes')
        .update({ status: quote.status })
        .eq('id', data.quote_id);
      
      return NextResponse.json(
        { error: 'Failed to create booking' },
        { status: 500 }
      );
    }
    
    // 5. Notify provider (via email queue)
    await emailQueue.add(
      'booking-created',
      {
        type: 'booking_created',
        data: {
          providerId: quote.provider_id,
          bookingId: booking.id,
          bookingReference: booking.booking_reference,
          rfqTitle: quote.service_request.title,
        },
      },
      defaultJobOptions
    );
    
    return NextResponse.json({
      success: true,
      booking: {
        id: booking.id,
        booking_reference: booking.booking_reference,
        status: booking.status,
        scheduled_start_date: booking.scheduled_start_date,
      },
      processing_time_ms: Date.now() - startTime,
    });
    
  } catch (error) {
    if (error instanceof z.ZodError) {
      return NextResponse.json(
        { error: 'Invalid request data', details: error.errors },
        { status: 400 }
      );
    }
    
    console.error('Quote acceptance error:', error);
    return NextResponse.json(
      { error: 'Internal server error' },
      { status: 500 }
    );
  }
}

typescriptimport { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { Redis } from '@upstash/redis';

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!,
});

export async function GET(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    const { searchParams } = new URL(request.url);
    const status = searchParams.get('status');
    const page = parseInt(searchParams.get('page') || '1');
    const perPage = 20;
    
    // Check cache
    const cacheKey = `rfq:list:${user.id}:${status}:${page}`;
    const cached = await redis.get(cacheKey);
    
    if (cached) {
      return NextResponse.json({
        ...cached,
        cached: true,
        query_time_ms: Date.now() - startTime,
      });
    }
    
    // Build query
    let query = supabase
      .from('service_requests')
      .select(`
        id,
        service_category,
        title,
        description,
        property_postcode,
        status,
        urgency_level,
        budget_min,
        budget_max,
        quote_count,
        created_at,
        expires_at
      `, { count: 'exact' })
      .eq('user_id', user.id)
      .order('created_at', { ascending: false });
    
    if (status) {
      query = query.eq('status', status);
    }
    
    const offset = (page - 1) * perPage;
    query = query.range(offset, offset + perPage - 1);
    
    const { data: rfqs, error, count } = await query;
    
    if (error) {
      throw error;
    }
    
    // For each RFQ, get quote count
    const rfqsWithQuotes = await Promise.all(
      (rfqs || []).map(async (rfq) => {
        const { count: quoteCount } = await supabase
          .from('quotes')
          .select('*', { count: 'exact', head: true })
          .eq('service_request_id', rfq.id)
          .in('status', ['sent', 'viewed', 'accepted']);
        
        return {
          ...rfq,
          quote_count: quoteCount || 0,
        };
      })
    );
    
    const response = {
      rfqs: rfqsWithQuotes,
      pagination: {
        page,
        per_page: perPage,
        total: count || 0,
        total_pages: Math.ceil((count || 0) / perPage),
      },
      query_time_ms: Date.now() - startTime,
    };
    
    // Cache for 60 seconds
    await redis.setex(cacheKey, 60, response);
    
    return NextResponse.json(response);
    
  } catch (error) {
    console.error('RFQ list error:', error);
    return NextResponse.json(
      { error: 'Failed to fetch service requests' },
      { status: 500 }
    );
  }
}

typescriptimport { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { Redis } from '@upstash/redis';

const redis = new Redis({
  url: process.env.UPSTASH_REDIS_URL!,
  token: process.env.UPSTASH_REDIS_TOKEN!,
});

export async function GET(request: NextRequest) {
  const startTime = Date.now();
  
  try {
    const supabase = await createClient();
    const { data: { user }, error: authError } = await supabase.auth.getUser();
    
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    const { searchParams } = new URL(request.url);
    const status = searchParams.get('status');
    const page = parseInt(searchParams.get('page') || '1');
    const perPage = 20;
    
    // Check cache
    const cacheKey = `quotes:list:${user.id}:${status}:${page}`;
    const cached = await redis.get(cacheKey);
    
    if (cached) {
      return NextResponse.json({
        ...cached,
        cached: true,
        query_time_ms: Date.now() - startTime,
      });
    }
    
    // Build query
    let query = supabase
      .from('quotes')
      .select(`
        id,
        quote_number,
        total_amount,
        status,
        sent_at,
        accepted_at,
        valid_until,
        service_request:service_requests!inner(
          id,
          title,
          service_category,
          property_postcode,
          urgency_level
        )
      `, { count: 'exact' })
      .eq('provider_id', user.id)
      .order('sent_at', { ascending: false });
    
    if (status) {
      query = query.eq('status', status);
    }
    
    const offset = (page - 1) * perPage;
    query = query.range(offset, offset + perPage - 1);
    
    const { data: quotes, error, count } = await query;
    
    if (error) {
      throw error;
    }
    
    const response = {
      quotes: quotes || [],
      pagination: {
        page,
        per_page: perPage,
        total: count || 0,
        total_pages: Math.ceil((count || 0) / perPage),
      },
      stats: {
        pending: 0, // TODO: Calculate
        accepted: 0,
        total_value: 0,
      },
      query_time_ms: Date.now() - startTime,
    };
    
    // Cache for 30 seconds
    await redis.setex(cacheKey, 30, response);
    
    return NextResponse.json(response);
    
  } catch (error) {
    console.error('Quote list error:', error);
    return NextResponse.json(
      { error: 'Failed to fetch quotes' },
      { status: 500 }
    );
  }
}
